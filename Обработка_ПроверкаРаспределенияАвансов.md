&НаКлиенте
Процедура ВыполнитьРаспределение(Команда)
	ОбработатьОчередьРасчетаАвансов();
КонецПроцедуры

// Процедура вызывается регламентным заданием
&НаСервере
Процедура ОбработатьОчередьРасчетаАвансов()

	ВидДействия = Перечисления.БИТ_ВидыДействийОтложеннойОбработки.РасчетАвансов;

	ТЗ_ОбъектыДляОтложенной = БИТ_ОтложеннаяОбработкаОбъектов.ПолучитьОбъектыДляОтложеннойОбработки(ВидДействия);

	Если ТЗ_ОбъектыДляОтложенной.Количество() = 0 Тогда
		Возврат;
	КонецЕсли;

	Для Каждого ТекСтрока Из ТЗ_ОбъектыДляОтложенной Цикл

		ТекстСообщения  = "";
		ЗаполнениеВыполнено = Ложь;

		Попытка
			ДокументОбработкиСсылка = ТекСтрока.Ссылка;
			ДатаРегистрации          = ТекСтрока.ДатаРегистрации;

			ИНК_РаспределениеОплат.ВыполнитьОтложенныйРасчет(ДокументОбработкиСсылка);

			БИТ_ОтложеннаяОбработкаОбъектов.ЗаписатьРезультатОтложеннойОбработки(
				ДокументОбработкиСсылка, ВидДействия, ДатаРегистрации);

		Исключение
			ТекстСообщения = ТекстСообщения + ОписаниеОшибки();
			БИТ_ОтложеннаяОбработкаОбъектов.ЗаписатьРезультатОтложеннойОбработки(
				ДокументОбработкиСсылка, ВидДействия, ДатаРегистрации, ТекстСообщения);
		КонецПопытки;

	КонецЦикла;

КонецПроцедуры

&НаКлиенте
Процедура ВыполнитьФиксациюМесяца(Команда)

	ПериодНачало = СтандПериод.ДатаНачала;
	ПериодКонец  = СтандПериод.ДатаОкончания;

	Если ПериодНачало >= ПериодКонец Тогда
		Сообщить("Некорректный период: дата начала должна быть меньше даты окончания.");
		Возврат;
	КонецЕсли;

	ТекстПодтверждения = СтрШаблон(
		"Выполнить закрытие периода %1 — %2?%3Все авто-записи распределений по оплатам периода будут зафиксированы и не смогут быть изменены автоматически.",
		Формат(ПериодНачало, "ДФ=дд.ММ.гггг"),
		Формат(ПериодКонец,  "ДФ=дд.ММ.гггг"),
		Символы.ПС);

	Если Вопрос(ТекстПодтверждения, РежимДиалогаВопрос.ДаНет) <> КодВозвратаДиалога.Да Тогда
		Возврат;
	КонецЕсли;

	Если НЕ ЗначениеЗаполнено(ЗаказКлиента) Тогда
		ВыполнитьФиксациюМесяцаНаСервере(Неопределено, ПериодНачало, ПериодКонец);
	Иначе
		ВыполнитьФиксациюМесяцаНаСервере(ЗаказКлиента, ПериодНачало, ПериодКонец);
	КонецЕсли;

КонецПроцедуры

&НаСервере
Процедура ВыполнитьФиксациюМесяцаНаСервере(ЗаказКлиента, ПериодНачало, ПериодКонец)

	Попытка
		ИНК_РаспределениеОплат.ЗакрытьПериодПоОплатам(ЗаказКлиента, ПериодНачало, ПериодКонец);
		Сообщить(СтрШаблон(
			"Закрытие периода %1 — %2 выполнено успешно.",
			Формат(ПериодНачало, "ДФ=дд.ММ.гггг"),
			Формат(ПериодКонец,  "ДФ=дд.ММ.гггг")));
	Исключение
		ТекстОшибки = ОписаниеОшибки();
		ЗаписьЖурналаРегистрации(
			"ИНК.ЗакрытиеПериодаОплат",
			УровеньЖурналаРегистрации.Ошибка, , , ТекстОшибки);
		Сообщить("Ошибка при закрытии периода: " + ТекстОшибки, СтатусСообщения.Важное);
	КонецПопытки;

КонецПроцедуры

// Команда кнопки «Загрузить переизбытки».
// Предупреждает, если в таблице есть несохранённые изменения.
//
&НаКлиенте
Процедура ЗагрузитьПереизбытки(Команда)

	// Проверяем наличие несохранённых изменений в таблице
	ЕстьИзменения = Ложь;
	Для Каждого Стр Из ТаблицаПереизбытков Цикл
		Если Стр.Обработать Тогда
			ЕстьИзменения = Истина;
			Прервать;
		КонецЕсли;
	КонецЦикла;

	Если ЕстьИзменения Тогда
		Ответ = Вопрос(
			"В таблице есть несохранённые изменения. Перезагрузить и потерять их?",
			РежимДиалогаВопрос.ДаНет);
		Если Ответ <> КодВозвратаДиалога.Да Тогда
			Возврат;
		КонецЕсли;
	КонецЕсли;

	ФильтрЗаказ = ?(ЗначениеЗаполнено(ЗаказКлиента), ЗаказКлиента, Неопределено);
	ЗагрузитьПереизбыткиНаСервере(ФильтрЗаказ);

КонецПроцедуры

// Серверная процедура загрузки.
// Один запрос к регистру — без циклов с запросами.
//
// Отбор:
//   Зафиксировано  = ЛОЖЬ
//   ЭтоПереизбыток = ИСТИНА
//   (Реализация = ПустаяСсылка) ИЛИ (ВыгруженоВБП = ЛОЖЬ)
//   ЗаказКлиента = &Заказ (опционально)
//
&НаСервере
Процедура ЗагрузитьПереизбыткиНаСервере(ФильтрЗаказ, РежимОтбора = "ннБезРеализации")

Запрос = Новый Запрос;
Запрос.Текст =
"ВЫБРАТЬ
|	РС.ЗаказКлиента КАК ЗаказКлиента,
|	РС.ДокументОплаты КАК ДокументОплаты,
|	РС.Реализация КАК Реализация,
|	РС.УИДЭтапа КАК УИДЭтапа,
|	РС.Сумма КАК СуммаПереизбытка,
|	РС.ТипОперации КАК ТипОперации,
|	ЕСТЬNULL(Этапы.ПроцентПлатежа, 0) КАК ПроцентЭтапа,
|	ЕСТЬNULL(ПРЕДСТАВЛЕНИЕССЫЛКИ(Этапы.ВариантОплаты), """") КАК НазваниеЭтапа
|ИЗ
|	РегистрСведений.ИНК_РаспределениеОплатПоЭтапам КАК РС
|		ЛЕВОЕ СОЕДИНЕНИЕ Справочник.ДоговорыКонтрагентов КАК Договор
|		ПО (Договор.Ссылка = РС.ЗаказКлиента.Договор)
|		ЛЕВОЕ СОЕДИНЕНИЕ Справочник.ДоговорыКонтрагентов.ИНК_ЭтапыГрафикаОплаты КАК Этапы
|		ПО (Этапы.Ссылка = Договор.Ссылка)
|			И (Этапы.УИДЭтапа = РС.УИДЭтапа)
|ГДЕ
|	РС.Зафиксировано = ЛОЖЬ
|	И РС.ЭтоПереизбыток = ИСТИНА
|	И (&ТолькоБезРеализации = ИСТИНА
|				И РС.Реализация = ЗНАЧЕНИЕ(Документ.РеализацияТоваровУслуг.ПустаяСсылка)
|			ИЛИ &ТолькоБезРеализации = ЛОЖЬ
|				И (РС.Реализация = ЗНАЧЕНИЕ(Документ.РеализацияТоваровУслуг.ПустаяСсылка)
|					ИЛИ РС.ВыгруженоВБП = ЛОЖЬ))
|	И (&ФильтрПоЗаказу = ЛОЖЬ
|			ИЛИ РС.ЗаказКлиента = &ЗаказКлиента)
|
|УПОРЯДОЧИТЬ ПО
|	РС.ЗаказКлиента,
|	РС.ДокументОплаты";  

ЕстьФильтрПоЗаказу = ЗначениеЗаполнено(ФильтрЗаказ);
Запрос.УстановитьПараметр("ФильтрПоЗаказу", ЕстьФильтрПоЗаказу);
Запрос.УстановитьПараметр("ЗаказКлиента",
	?(ЕстьФильтрПоЗаказу, ФильтрЗаказ, Документы.ЗаказКлиента.ПустаяСсылка()));
// Признак режима — Истина: только без реализации; Ложь: все незафиксированные переизбытки
Запрос.УстановитьПараметр("ТолькоБезРеализации", РежимОтбора = "БезРеализации");

	Выборка = Запрос.Выполнить().Выбрать();

	ТаблицаПереизбытков.Очистить();

	Пока Выборка.Следующий() Цикл
		НовСтр = ТаблицаПереизбытков.Добавить();
		НовСтр.Обработать             = Ложь;
		НовСтр.ЗаказКлиента           = Выборка.ЗаказКлиента;
		НовСтр.ДокументОплаты         = Выборка.ДокументОплаты;
		НовСтр.Реализация              = Выборка.Реализация;
		НовСтр.СуммаПереизбытка        = Выборка.СуммаПереизбытка;
		НовСтр.СуммаФиксации           = 0;
		НовСтр.МаксДоступнаяСумма      = 0;
		
		НовСтр.УИДЭтапаФиксации  = Выборка.УИДЭтапа;
		НовСтр.НазваниеЭтапа     = Выборка.НазваниеЭтапа;
		НовСтр.ПроцентЭтапа      = Выборка.ПроцентЭтапа;
		НовСтр.УИДЭтапаИсходный  = Выборка.УИДЭтапа; // фиксируем исходное значение
		
		
		НовСтр.ТипОперации             = Выборка.ТипОперации;
		НовСтр.ЭтоНоваяСтрока         = Ложь;
		// Исходный ключ — для поиска записи в НаборЗаписей за O(1)
		НовСтр.ИсходныйКлюч           = СформироватьИсходныйКлюч(
			Выборка.ЗаказКлиента,
			Выборка.ДокументОплаты,
			Выборка.УИДЭтапа,
			Выборка.Реализация);
		// Сохраняем исходные значения для определения факта изменения
		НовСтр.РеализацияИсходная      = Выборка.Реализация;
		НовСтр.СуммаФиксацииИсходная   = 0;
	КонецЦикла;

КонецПроцедуры

// Обработчик кнопки выбора этапа в строке таблицы.
// Получает список этапов договора, показывает пользователю, заполняет строку.
//
&НаКлиенте
Процедура ТаблицаПереизбытковУИДЭтапаФиксацииНачалоВыбора(Элемент, ДанныеВыбора, ВыборДобавлением, СтандартнаяОбработка)
		СтандартнаяОбработка = Ложь;
	 
	 
	ТекСтрока = Элементы.ТаблицаПереизбытков.ТекущиеДанные;
	Если ТекСтрока = Неопределено Тогда
		Возврат;
	КонецЕсли;

	Если НЕ ЗначениеЗаполнено(ТекСтрока.ЗаказКлиента) Тогда
		Сообщить("Строка не содержит заказа клиента.", СтатусСообщения.Важное);
		Возврат;
	КонецЕсли;

	// Один серверный вызов — получаем список этапов
	МассивЭтапов = ПолучитьЭтапыДоговораПоЗаказу(ТекСтрока.ЗаказКлиента);

	Если МассивЭтапов = Неопределено ИЛИ МассивЭтапов.Количество() = 0 Тогда
		Сообщить("По договору заказа не найдены этапы графика оплаты.", СтатусСообщения.Важное);
		Возврат;
	КонецЕсли;

	// Формируем список для выбора
	СписокЭтапов = Новый СписокЗначений;
	Для Каждого Эт Из МассивЭтапов Цикл
		Представление = СтрШаблон("%1 (%2) %3 %%",
			Эт.НомерСтроки,  
			Строка(Эт.ВариантОплаты),
			Формат(Эт.ПроцентПлатежа, "ЧДЦ=0"));
		СписокЭтапов.Добавить(Эт, Представление);
	КонецЦикла;

	ВыбранныйЭлемент = СписокЭтапов.ВыбратьЭлемент("Выберите этап для фиксации");
	Если ВыбранныйЭлемент = Неопределено Тогда
		Возврат;
	КонецЕсли;

	ВыбранныйЭтап = ВыбранныйЭлемент.Значение;
	
	ПредставлениеЭтапа = СтрШаблон("%1 (%2) %3 %%",
			ВыбранныйЭтап.НомерСтроки,  
			Строка(ВыбранныйЭтап.ВариантОплаты),
			Формат(ВыбранныйЭтап.ПроцентПлатежа, "ЧДЦ=0"));

	
	// Заполняем строку таблицы — без дополнительного серверного вызова
	ТекСтрока.УИДЭтапаФиксации = ВыбранныйЭтап.УИДЭтапа;
	ТекСтрока.НазваниеЭтапа    = ПредставлениеЭтапа;
	ТекСтрока.ПроцентЭтапа     = ВыбранныйЭтап.ПроцентПлатежа;

	// Если реализация уже выбрана — пересчитываем лимит
	ОбновитьМаксДоступнуюСумму(ТекСтрока);

	// Устанавливаем флаг изменения
	УстановитьФлагОбработать(ТекСтрока);

КонецПроцедуры

// Получает список этапов графика оплаты из договора заказа.
// Договор берётся через BSP-метод — без обращения через точку (анти-паттерн #2).
//
&НаСервереБезКонтекста
Функция ПолучитьЭтапыДоговораПоЗаказу(ЗаказКлиента)

	// Получаем договор одним точечным запросом через БСП
	Договор = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ЗаказКлиента, "Договор");

	Если НЕ ЗначениеЗаполнено(Договор) Тогда
		Возврат Новый Массив;
	КонецЕсли;

	// Получаем этапы одним запросом
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ
	|	Этапы.УИДЭтапа КАК УИДЭтапа,
	|	Этапы.ПроцентПлатежа КАК ПроцентПлатежа,
	|	Этапы.ВариантОплаты КАК ВариантОплаты,
	|	Этапы.НомерСтроки КАК НомерСтроки
	|ИЗ
	|	Справочник.ДоговорыКонтрагентов.ИНК_ЭтапыГрафикаОплаты КАК Этапы
	|ГДЕ
	|	Этапы.Ссылка = &Договор
	|
	|УПОРЯДОЧИТЬ ПО
	|	Этапы.НомерСтроки";

	Запрос.УстановитьПараметр("Договор", Договор);

	РезультатЗапроса = Запрос.Выполнить();

	Если РезультатЗапроса.Пустой() Тогда
		Возврат Новый Массив;
	КонецЕсли;

	// Выгружаем в массив структур (передаём на клиент)
	МассивЭтапов = Новый Массив;
	Выборка = РезультатЗапроса.Выбрать();
	Пока Выборка.Следующий() Цикл
		Эт = Новый Структура(
			"УИДЭтапа, ПроцентПлатежа, ВариантОплаты, НомерСтроки",
			Выборка.УИДЭтапа,
			Выборка.ПроцентПлатежа,
			Выборка.ВариантОплаты,
			Выборка.НомерСтроки);
		МассивЭтапов.Добавить(Эт);
	КонецЦикла;

	Возврат МассивЭтапов;

Процедура УстановитьФлагОбработать(Стр)

	Если Стр.ЭтоНоваяСтрока Тогда
		Стр.Обработать = Истина;
		Возврат;
	КонецЕсли;

	// Сравниваем с исходными значениями
	Изменено = (Стр.Реализация         <> Стр.РеализацияИсходная)
		ИЛИ (Стр.УИДЭтапаФиксации  <> Стр.УИДЭтапаИсходный)

//──────────────────────────────────────────────────────────────────────────────

//──────────────────────────────────────────────────────────────────────────────

// Вызывается с клиента после изменения этапа или реализации.
// Если оба поля заполнены — делает один серверный вызов.
//
&НаКлиенте
Процедура ОбновитьМаксДоступнуюСумму(Стр)

	Если НЕ ЗначениеЗаполнено(Стр.УИДЭтапаФиксации)
		ИЛИ НЕ ЗначениеЗаполнено(Стр.Реализация) Тогда
		// Оба поля ещё не выбраны — не считаем
		Стр.МаксДоступнаяСумма = 0;
		Возврат;
	КонецЕсли;

	// Параметры для серверного вызова
	ПараметрыРасчета = Новый Структура;
	ПараметрыРасчета.Вставить("ЗаказКлиента",     Стр.ЗаказКлиента);
	ПараметрыРасчета.Вставить("УИДЭтапа",         Стр.УИДЭтапаФиксации);
	ПараметрыРасчета.Вставить("Реализация",       Стр.Реализация);
	ПараметрыРасчета.Вставить("ДокументОплаты",   Стр.ДокументОплаты);
	ПараметрыРасчета.Вставить("СуммаПереизбытка", Стр.СуммаПереизбытка);
	ПараметрыРасчета.Вставить("ПроцентЭтапа",     Стр.ПроцентЭтапа);

	Результат = ПересчитатьМаксДоступнуюСумму(ПараметрыРасчета);

	Стр.МаксДоступнаяСумма = Результат.МаксДоступнаяСумма;

	// Предзаполняем СуммаФиксации, если пользователь ещё не вводил вручную
	Если Стр.СуммаФиксации = 0 И Результат.МаксДоступнаяСумма > 0 Тогда
		Стр.СуммаФиксации = Результат.МаксДоступнаяСумма;
	КонецЕсли;

	// Информируем пользователя о лимите
	Если Результат.МаксДоступнаяСумма <= 0 Тогда
		Сообщить(СтрШаблон(
			"Лимит по этапу «%1» / реализации [%2] исчерпан. Фиксация невозможна.",
			Стр.НазваниеЭтапа,
			Формат(Стр.Реализация, "П=Пустая ссылка")),
			СтатусСообщения.Внимание);
	Иначе
		Сообщить(СтрШаблон(
			"Доступно к фиксации: %1 руб. (этап «%2»)",
			Формат(Результат.МаксДоступнаяСумма, "ЧДЦ=2; ЧРД=."),
			Стр.НазваниеЭтапа));
	КонецЕсли;

// Серверная функция расчёта максимально доступной суммы фиксации.
// Один пакетный запрос — лимит этапа + лимит РТУ в этапе.
//
// Параметры:
//   Параметры — Структура:
//     ЗаказКлиента    — ДокументСсылка.ЗаказКлиента
//     УИДЭтапа        — Строка — выбранный этап фиксации
//     Реализация      — ДокументСсылка.РеализацияТоваровУслуг
//     ДокументОплаты  — ДокументСсылка (для исключения самой записи-переизбытка)
//     СуммаПереизбытка — Число — максимум из самой записи
//     ПроцентЭтапа    — Число — % этапа (для расчёта лимита этапа)
//
// Возвращает: Структура { МаксДоступнаяСумма }
//
&НаСервереБезКонтекста
Функция ПересчитатьМаксДоступнуюСумму(Параметры)

	Результат = Новый Структура("МаксДоступнаяСумма", 0);

	// Получаем СуммуДокумента заказа и СуммуДокумента реализации одним запросом
	// Заказ и реализация — разные типы, поэтому два запроса объединяем пакетом
	МенеджерВТ = Новый МенеджерВременныхТаблиц;
	Запрос = Новый Запрос;
	Запрос.МенеджерВременныхТаблиц = МенеджерВТ;

	// Пакет: 1) суммы заказа и реализации; 2) уже распределено по этапу; 3) уже распределено по этапу+РТУ
	Запрос.Текст =
	// --- Блок 1: сумма заказа и сумма реализации ---
	"ВЫБРАТЬ
	|	Заказ.СуммаДокумента КАК СуммаЗаказа
	|ПОМЕСТИТЬ ВТ_СуммаЗаказа
	|ИЗ
	|	Документ.ЗаказКлиента КАК Заказ
	|ГДЕ
	|	Заказ.Ссылка = &ЗаказКлиента
	|;
	|ВЫБРАТЬ
	|	РТУ.СуммаДокумента КАК СуммаРТУ
	|ПОМЕСТИТЬ ВТ_СуммаРТУ
	|ИЗ
	|	Документ.РеализацияТоваровУслуг КАК РТУ
	|ГДЕ
	|	РТУ.Ссылка = &Реализация
	|;
	|// --- Блок 2: уже распределено по этапу (все РТУ, кроме самой записи-переизбытка) ---
	|ВЫБРАТЬ
	|	ЕСТЬNULL(СУММА(РС.Сумма), 0) КАК УжеПоЭтапу
	|ПОМЕСТИТЬ ВТ_РаспрПоЭтапу
	|ИЗ
	|	РегистрСведений.ИНК_РаспределениеОплатПоЭтапам КАК РС
	|ГДЕ
	|	РС.ЗаказКлиента = &ЗаказКлиента
	|	И РС.УИДЭтапа   = &УИДЭтапа
	|	И НЕ (РС.ДокументОплаты  = &ДокументОплаты
	|		   И РС.ЭтоПереизбыток = ИСТИНА
	|		   И РС.Зафиксировано  = ЛОЖЬ)
	|;
	|// --- Блок 3: уже распределено по паре этап+РТУ (кроме самой записи-переизбытка) ---
	|ВЫБРАТЬ
	|	ЕСТЬNULL(СУММА(РС.Сумма), 0) КАК УжеПоЭтапуРТУ
	|ПОМЕСТИТЬ ВТ_РаспрПоЭтапуРТУ
	|ИЗ
	|	РегистрСведений.ИНК_РаспределениеОплатПоЭтапам КАК РС
	|ГДЕ
	|	РС.ЗаказКлиента = &ЗаказКлиента
	|	И РС.УИДЭтапа   = &УИДЭтапа
	|	И РС.Реализация  = &Реализация
	|	И НЕ (РС.ДокументОплаты  = &ДокументОплаты
	|		   И РС.ЭтоПереизбыток = ИСТИНА
	|		   И РС.Зафиксировано  = ЛОЖЬ)
	|;
	|// --- Итог: собираем все данные в одну строку ---
	|ВЫБРАТЬ
	|	ВТ_СуммаЗаказа.СуммаЗаказа    КАК СуммаЗаказа,
	|	ВТ_СуммаРТУ.СуммаРТУ          КАК СуммаРТУ,
	|	ВТ_РаспрПоЭтапу.УжеПоЭтапу    КАК УжеПоЭтапу,
	|	ВТ_РаспрПоЭтапуРТУ.УжеПоЭтапуРТУ КАК УжеПоЭтапуРТУ
	|ИЗ
	|	ВТ_СуммаЗаказа КАК ВТ_СуммаЗаказа,
	|	ВТ_СуммаРТУ    КАК ВТ_СуммаРТУ,
	|	ВТ_РаспрПоЭтапу КАК ВТ_РаспрПоЭтапу,
	|	ВТ_РаспрПоЭтапуРТУ КАК ВТ_РаспрПоЭтапуРТУ";

	Запрос.УстановитьПараметр("ЗаказКлиента",   Параметры.ЗаказКлиента);
	Запрос.УстановитьПараметр("Реализация",      Параметры.Реализация);
	Запрос.УстановитьПараметр("УИДЭтапа",        Параметры.УИДЭтапа);
	Запрос.УстановитьПараметр("ДокументОплаты",  Параметры.ДокументОплаты);

	РезультатПакета = Запрос.ВыполнитьПакет();
	// Последний результат пакета — итоговая выборка
	Выборка = РезультатПакета[РезультатПакета.ВГраница()].Выбрать();

	Если НЕ Выборка.Следующий() Тогда
		Возврат Результат;
	КонецЕсли;

	СуммаЗаказа  = Выборка.СуммаЗаказа;
	СуммаРТУ     = Выборка.СуммаРТУ;
	УжеПоЭтапу   = Выборка.УжеПоЭтапу;
	УжеПоРТУЭтап = Выборка.УжеПоЭтапуРТУ;

	// Лимит этапа = СуммаЗаказа × Процент / 100
	СуммаЭтапа = Окр(СуммаЗаказа * Параметры.ПроцентЭтапа / 100, 2);

	МаксПоЭтапу = СуммаЭтапа   - УжеПоЭтапу;   // свободный остаток лимита этапа
	МаксПоРТУ   = СуммаРТУ     - УжеПоРТУЭтап; // свободный остаток лимита РТУ в этапе

	// Итоговый максимум — наименьшее из трёх ограничений
	МаксДоступная = Мин(МаксПоЭтапу, МаксПоРТУ, Параметры.СуммаПереизбытка);

	Результат.МаксДоступнаяСумма = ?(МаксДоступная < 0, 0, МаксДоступная);

	Возврат Результат;

КонецФункции


//──────────────────────────────────────────────────────────────────────────────
// 2.5. ФИКСАЦИЯ ОТМЕЧЕННЫХ СТРОК
//──────────────────────────────────────────────────────────────────────────────

// Команда кнопки «Зафиксировать отмеченные».
// Валидация на клиенте → серверная фиксация → перезагрузка таблицы.
//
&НаКлиенте
Процедура ЗафиксироватьОтмеченные(Команда)

	// Собираем строки для обработки
	СтрокиДляФиксации = Новый Массив;
	Для Каждого Стр Из ТаблицаПереизбытков Цикл
		Если Стр.Обработать Тогда
			СтрокиДляФиксации.Добавить(Стр);
		КонецЕсли;
	КонецЦикла;

	Если СтрокиДляФиксации.Количество() = 0 Тогда
		Сообщить("Нет отмеченных строк для фиксации.");
		Возврат;
	КонецЕсли;

	// Валидация
	ТекстыОшибок = Новый Массив;
	НомерСтрФормы = 0;
	Для Каждого Стр Из ТаблицаПереизбытков Цикл
		НомерСтрФормы = НомерСтрФормы + 1;
		Если НЕ Стр.Обработать Тогда
			Продолжить;
		КонецЕсли;

		Если НЕ ЗначениеЗаполнено(Стр.УИДЭтапаФиксации) Тогда
			ТекстыОшибок.Добавить(СтрШаблон("Строка %1: не выбран этап для фиксации.", НомерСтрФормы));
		КонецЕсли;

		Если НЕ ЗначениеЗаполнено(Стр.Реализация) Тогда
			ТекстыОшибок.Добавить(СтрШаблон("Строка %1: не выбрана реализация.", НомерСтрФормы));
		КонецЕсли;

		Если Стр.СуммаФиксации <= 0 Тогда
			ТекстыОшибок.Добавить(СтрШаблон("Строка %1: сумма фиксации должна быть больше нуля.", НомерСтрФормы));
		КонецЕсли;

		Если Стр.СуммаФиксации > Стр.СуммаПереизбытка Тогда
			ТекстыОшибок.Добавить(СтрШаблон(
				"Строка %1: сумма фиксации (%2) превышает сумму переизбытка (%3).",
				НомерСтрФормы,
				Формат(Стр.СуммаФиксации,   "ЧДЦ=2"),
				Формат(Стр.СуммаПереизбытка, "ЧДЦ=2")));
		КонецЕсли;

		Если Стр.МаксДоступнаяСумма > 0 И Стр.СуммаФиксации > Стр.МаксДоступнаяСумма Тогда
			ТекстыОшибок.Добавить(СтрШаблон(
				"Строка %1: сумма фиксации (%2) превышает доступный лимит (%3).",
				НомерСтрФормы,
				Формат(Стр.СуммаФиксации,      "ЧДЦ=2"),
				Формат(Стр.МаксДоступнаяСумма,  "ЧДЦ=2")));
		КонецЕсли;

	КонецЦикла;

	Если ТекстыОшибок.Количество() > 0 Тогда
		Сообщить(СтрСоединить(ТекстыОшибок, Символы.ПС), СтатусСообщения.Важное);
		Возврат;
	КонецЕсли;

	// Подтверждение
	Если Вопрос(
		СтрШаблон("Зафиксировать %1 строк(у) переизбытка?", СтрокиДляФиксации.Количество()),
		РежимДиалогаВопрос.ДаНет) <> КодВозвратаДиалога.Да Тогда
		Возврат;
	КонецЕсли;

	// Передаём на сервер
	ЗафиксироватьПереизбыткиНаСервере();

КонецПроцедуры

// Серверная процедура фиксации.
// Группировка по заказам → одна блокировка + транзакция на заказ.
// Без запросов в цикле: набор записей читается целиком, индексируется за O(n),
// далее поиск за O(1) по ключу.
//
&НаСервере
Процедура ЗафиксироватьПереизбыткиНаСервере()

	// Собираем отмеченные строки из реквизита формы
	ОтмеченныеСтроки = ТаблицаПереизбытков.НайтиСтроки(Новый Структура("Обработать", Истина));

	Если ОтмеченныеСтроки.Количество() = 0 Тогда
		Возврат;
	КонецЕсли;

	// Группируем по ЗаказКлиента — Соответствие: Заказ -> Массив строк
	СтрокиПоЗаказам = Новый Соответствие;
	Для Каждого Стр Из ОтмеченныеСтроки Цикл
		МассивЗаказа = СтрокиПоЗаказам[Стр.ЗаказКлиента];
		Если МассивЗаказа = Неопределено Тогда
			МассивЗаказа = Новый Массив;
			СтрокиПоЗаказам.Вставить(Стр.ЗаказКлиента, МассивЗаказа);
		КонецЕсли;
		МассивЗаказа.Добавить(Стр);
	КонецЦикла;

	// Накапливаем ошибки — не прерываем весь пакет из-за одного заказа
	ТекстыОшибок = Новый Массив;

	Для Каждого ЭлЗаказ Из СтрокиПоЗаказам Цикл

		ТекЗаказ    = ЭлЗаказ.Ключ;
		СтрокиЗаказа = ЭлЗаказ.Значение;

		Блокировка = Новый БлокировкаДанных;
		Элемент = Блокировка.Добавить("РегистрСведений.ИНК_РаспределениеОплатПоЭтапам");
		Элемент.УстановитьЗначение("ЗаказКлиента", ТекЗаказ);
		Элемент.Режим = РежимБлокировкиДанных.Исключительный;

		НачатьТранзакцию();
		Попытка

			Блокировка.Заблокировать();

			// Читаем весь набор записей по заказу одним вызовом
			Набор = РегистрыСведений.ИНК_РаспределениеОплатПоЭтапам.СоздатьНаборЗаписей();
			Набор.Отбор.ЗаказКлиента.Установить(ТекЗаказ);
			Набор.Прочитать();

			// Строим индекс существующих записей за O(n) — ключ: ИсходныйКлюч
			ИндексЗаписей = Новый Соответствие;
			Для Каждого ЗапРС Из Набор Цикл
				КлючЗаписи = СформироватьИсходныйКлюч(
					ЗапРС.ЗаказКлиента,
					ЗапРС.ДокументОплаты,
					ЗапРС.УИДЭтапа,
					ЗапРС.Реализация);
				ИндексЗаписей.Вставить(КлючЗаписи, ЗапРС);
			КонецЦикла;

			// Строки, помеченные к удалению из набора (полная выборка суммы)
			КлючиДляУдаления = Новый Соответствие;

			// Новые записи, которые добавим в набор
			НовыеЗаписи = Новый Массив;

			Для Каждого Стр Из СтрокиЗаказа Цикл

				// Повторная проверка лимита на сервере внутри транзакции
				ПарамЛимит = Новый Структура;
				ПарамЛимит.Вставить("ЗаказКлиента",    Стр.ЗаказКлиента);
				ПарамЛимит.Вставить("УИДЭтапа",        Стр.УИДЭтапаФиксации);
				ПарамЛимит.Вставить("Реализация",      Стр.Реализация);
				ПарамЛимит.Вставить("ДокументОплаты",  Стр.ДокументОплаты);
				ПарамЛимит.Вставить("СуммаПереизбытка", Стр.СуммаПереизбытка);
				ПарамЛимит.Вставить("ПроцентЭтапа",    Стр.ПроцентЭтапа);

				РезЛимит = ПересчитатьМаксДоступнуюСумму(ПарамЛимит);

				Если РезЛимит.МаксДоступнаяСумма <= 0 Тогда
					ВызватьИсключение СтрШаблон(
						"Лимит по этапу «%1» / реализации исчерпан (заказ %2). Фиксация отменена.",
						Стр.НазваниеЭтапа, ТекЗаказ);
				КонецЕсли;

				Если Стр.СуммаФиксации > РезЛимит.МаксДоступнаяСумма Тогда
					ВызватьИсключение СтрШаблон(
						"Сумма фиксации (%1) превышает актуальный лимит (%2) по этапу «%3» (заказ %4).",
						Формат(Стр.СуммаФиксации,              "ЧДЦ=2"),
						Формат(РезЛимит.МаксДоступнаяСумма,   "ЧДЦ=2"),
						Стр.НазваниеЭтапа,
						ТекЗаказ);
				КонецЕсли;

				// Находим исходную запись-переизбыток за O(1)
				ИсходнаяЗапись = ИндексЗаписей[Стр.ИсходныйКлюч];

				Если ИсходнаяЗапись = Неопределено Тогда
					ВызватьИсключение СтрШаблон(
						"Исходная запись переизбытка не найдена (ключ: %1). Возможно, данные изменились. Перезагрузите таблицу.",
						Стр.ИсходныйКлюч);
				КонецЕсли;

				// Обновляем или помечаем исходную запись к удалению
				Если Стр.СуммаФиксации < Стр.СуммаПереизбытка Тогда
					// Частичная фиксация — уменьшаем остаток в исходной записи
					ИсходнаяЗапись.Сумма = Стр.СуммаПереизбытка - Стр.СуммаФиксации;
				Иначе
					// Полная фиксация — исходную запись удаляем
					КлючиДляУдаления.Вставить(Стр.ИсходныйКлюч, Истина);
				КонецЕсли;

				// Формируем новую зафиксированную запись
				НовЗап = Новый Структура;
				НовЗап.Вставить("ЗаказКлиента",   Стр.ЗаказКлиента);
				НовЗап.Вставить("ДокументОплаты",  Стр.ДокументОплаты);
				НовЗап.Вставить("Реализация",      Стр.Реализация);
				НовЗап.Вставить("УИДЭтапа",        Стр.УИДЭтапаФиксации);
				НовЗап.Вставить("Зафиксировано",   Истина);
				НовЗап.Вставить("Сумма",           Стр.СуммаФиксации);
				НовЗап.Вставить("ЭтоПереизбыток",  Истина);
				НовЗап.Вставить("ТипОперации",     Стр.ТипОперации);
				НовЗап.Вставить("ВыгруженоВБП",    Ложь);
				НовыеЗаписи.Добавить(НовЗап);

			КонецЦикла;

			// Перестраиваем набор:
			// 1) Переносим все записи, кроме помеченных к удалению
			ТЗДляЗаписи = Набор.Выгрузить();

			// Удаляем помеченные строки из ТЗ (по ключу)
			// Строим индекс строк ТЗ для удаления — O(n)
			СтрокиТЗДляУдаления = Новый Массив;
			Для Каждого СтрТЗ Из ТЗДляЗаписи Цикл
				КлючТЗ = СформироватьИсходныйКлюч(
					СтрТЗ.ЗаказКлиента,
					СтрТЗ.ДокументОплаты,
					СтрТЗ.УИДЭтапа,
					СтрТЗ.Реализация);
				Если КлючиДляУдаления[КлючТЗ] <> Неопределено Тогда
					СтрокиТЗДляУдаления.Добавить(СтрТЗ);
				КонецЕсли;
			КонецЦикла;
			Для Каждого СтрУдал Из СтрокиТЗДляУдаления Цикл
				ТЗДляЗаписи.Удалить(СтрУдал);
			КонецЦикла;

			// 2) Добавляем новые зафиксированные записи
			Для Каждого НовЗап Из НовыеЗаписи Цикл
				НовСтрТЗ = ТЗДляЗаписи.Добавить();
				ЗаполнитьЗначенияСвойств(НовСтрТЗ, НовЗап);
			КонецЦикла;

			Набор.Загрузить(ТЗДляЗаписи);
			Набор.Записать(Истина);

			ЗафиксироватьТранзакцию();

		Исключение
			ОтменитьТранзакцию();
			ТекстОшибки = СтрШаблон(
				"Ошибка при фиксации переизбытка по заказу %1: %2",
				ТекЗаказ, ОписаниеОшибки());
			ЗаписьЖурналаРегистрации(
				"ИНК.ФиксацияПереизбытков",
				УровеньЖурналаРегистрации.Ошибка, , , ТекстОшибки);
			ТекстыОшибок.Добавить(ТекстОшибки);
		КонецПопытки;

	КонецЦикла;

	// Перезагружаем таблицу после всех операций
	ФильтрЗаказ = ?(ЗначениеЗаполнено(ЗаказКлиента), ЗаказКлиента, Неопределено);
	ЗагрузитьПереизбыткиНаСервере(ФильтрЗаказ);

	// Выводим накопленные ошибки
	Если ТекстыОшибок.Количество() > 0 Тогда
		Сообщить(
			"При фиксации возникли ошибки:" + Символы.ПС
			+ СтрСоединить(ТекстыОшибок, Символы.ПС),
			СтатусСообщения.Важное);
	Иначе
		Сообщить("Фиксация переизбытков выполнена успешно.");
	КонецЕсли;

КонецПроцедуры
