//
// Параметры:
//  ЗаказКлиента		 -  - 
//  ИсточникРасчета	 -  - 
//  ВерсияАлгоритма	 -  - 
//  Пользователь		 -  -

// Возвращаемое значение:
//   -

&НаСервере
Функция НачатьДиагностику(ЗаказКлиента, ИсточникРасчета, ВерсияАлгоритма, Пользователь = Неопределено) Экспорт
	Контекст = Новый Структура;
	Контекст.Вставить("ЗаказКлиента", ЗаказКлиента);
	Контекст.Вставить("ИсточникРасчета", ИсточникРасчета);
	Контекст.Вставить("ВерсияАлгоритма", ВерсияАлгоритма);
	Контекст.Вставить("Пользователь", Пользователь);

	Набор = РегистрыСведений.ИНК_ДиагностикаРаспределенияОплат.СоздатьНаборЗаписей();
	Набор.Отбор.ЗаказКлиента.Установить(ЗаказКлиента);
	Набор.Прочитать();

	Индекс = Новый Соответствие; // ключ: Код|КлючСитуации -> индекс строки в Набор.Записи

	Для Инд = 0 По Набор.Количество() - 1 Цикл
		Зап = Набор[Инд];

		// Закрываем все текущие записи (история сохраняется)
		Зап.Актуально = Ложь;

		КлючИндекса = СформироватьКлючИндекса(Зап.КодСитуации, Зап.КлючСитуации);
		// Если дубликаты уже есть, оставляем в индексе последнюю встреченную
		Индекс.Вставить(КлючИндекса, Инд);
	КонецЦикла;

	Контекст.Вставить("Набор", Набор);
	Контекст.Вставить("Индекс", Индекс);

	Возврат Контекст;
КонецФункции

&НаСервере
Процедура ЗафиксироватьСитуацию(Контекст, КодСитуации, Уровень, Сообщение, СуммаОтклонения = 0,
		УИДЭтапа = "", ДокументОплаты = Неопределено, Реализация = Неопределено, Детали = "") Экспорт

	Если Контекст = Неопределено Тогда
		Возврат;
	КонецЕсли;

	КлючСитуации = СформироватьКлючСитуации(УИДЭтапа, ДокументОплаты, Реализация);
	КлючИндекса  = СформироватьКлючИндекса(КодСитуации, КлючСитуации);

	Набор = Контекст.Набор;
	Индекс = Контекст.Индекс;
	
	ИндСтроки = Индекс.Получить(КлючИндекса);
	
	Если ИндСтроки <> Неопределено Тогда
		Зап = Набор[ИндСтроки];
		Зап.Количество = Зап.Количество + 1;
	Иначе
		Зап = Набор.Добавить();

		// Измерения
		Зап.ЗаказКлиента   = Контекст.ЗаказКлиента;
		Зап.КодСитуации    = КодСитуации;
		Зап.КлючСитуации   = КлючСитуации;
		Зап.УИДЭтапа       = УИДЭтапа;
		Зап.ДокументОплаты = ?(ЗначениеЗаполнено(ДокументОплаты), ДокументОплаты, Неопределено);
		Зап.Реализация     = ?(ЗначениеЗаполнено(Реализация), Реализация, Неопределено);

		// Ресурсы
		Зап.Количество      = 1;
		Зап.СуммаОтклонения = 0;

		Индекс.Вставить(КлючИндекса, Набор.Количество() - 1);
	КонецЕсли;

	// Обновляем общие поля (актуализируем)
	Зап.Уровень            = Уровень;
	Зап.Сообщение          = Лев(Сообщение, 500);
	Зап.СуммаОтклонения    = СуммаОтклонения;
	Зап.ДатаВремяФиксации  = ТекущаяДата();
	Зап.Актуально          = Истина;

	//	Зап.Пользователь = Контекст.Пользователь;
	//КонецЕсли;

Функция СформироватьКлючСитуации(УИДЭтапа, ДокументОплаты, Реализация)
	
	ЭтапКлюч = ?(ЗначениеЗаполнено(УИДЭтапа), УИДЭтапа, "-");
	ОплатаКлюч = ПредставлениеСсылкиДляКлюча(ДокументОплаты);
	РеализацияКлюч = ПредставлениеСсылкиДляКлюча(Реализация);

	Возврат СтрШаблон("Этап=%1;Оплата=%2;Реализация=%3", ЭтапКлюч, ОплатаКлюч, РеализацияКлюч);  
	
КонецФункции

Функция ПредставлениеСсылкиДляКлюча(Ссылка)
	
	Если Не ЗначениеЗаполнено(Ссылка) Тогда
		Возврат "-";
	КонецЕсли;

	// Уникальный идентификатор ссылки — стабильный ключ (36 символов)
	Возврат Строка(Ссылка.УникальныйИдентификатор());
