Функция ПолучитьТаблицуВедомостейНаСервереБезКонтекста(Знач Организация, ДатаДокумента)          
   
   УстановитьПривилегированныйРежим(Истина);
   
   
   ПодключениеHTTP = "";
   СерверHTTP = ""; 
   
   
   Запрос = Новый Запрос;
   Запрос.Текст = "ВЫБРАТЬ
                  |  ИНК_ПараметрыПодключенийКБазамДанных.ПодключениеHTTP КАК ПодключениеHTTP,
                  |  ИНК_ПараметрыПодключенийКБазамДанных.СерверHTTP КАК СерверHTTP
                  |ИЗ
                  |  РегистрСведений.ИНК_ПараметрыПодключенийКБазамДанных КАК ИНК_ПараметрыПодключенийКБазамДанных
                  |     ВНУТРЕННЕЕ СОЕДИНЕНИЕ Справочник.Организации КАК Организации
                  |     ПО ИНК_ПараметрыПодключенийКБазамДанных.ИНН = Организации.ИНН
                  |ГДЕ
                  |  Организации.Ссылка = &Организация";
   Запрос.УстановитьПараметр("Организация", Организация);
   Выборка = Запрос.Выполнить().Выбрать();
   
   Пока Выборка.Следующий() Цикл
    ПодключениеHTTP = Выборка.ПодключениеHTTP;
    СерверHTTP      = Выборка.СерверHTTP;
   КонецЦикла;
   
   Если ПодключениеHTTP <> "" И СерверHTTP <> "" Тогда
      
      
      Соединение = Новый HTTPСоединение(СерверHTTP,80); // пример win-sql-dev.inkotech.ru
      ЗапросHttp = Новый HTTPЗапрос(ПодключениеHTTP+"/SalaryStatement"+"?datefrom="+Формат(ДатаДокумента,"ДФ=yyyyMMdd"));   // Пример    "/INK_SYNC/hs/Buh/SalaryStatement";
      
      Ответ = Соединение.Получить(ЗапросHttp);
      
      Если Ответ.КодСостояния <> 200 Тогда
         СтруктураОтвета = Новый Структура;
         СтруктураОтвета.Вставить("Результат", "Ошибка запроса к базе");
         СтруктураОтвета.Вставить("ТЗ", Новый ТаблицаЗначений);
      Иначе
         
         
         ТелоОтвета = Ответ.ПолучитьТелоКакСтроку();
         
         ЧтениеJSON = Новый ЧтениеJSON;
         ЧтениеJSON.УстановитьСтроку(ТелоОтвета);
         ОбъектXDTO = СериализаторXDTO.ПрочитатьJSON(ЧтениеJSON, Тип("ТаблицаЗначений"));
         
         
         СтруктураОтвета = Новый Структура;
         СтруктураОтвета.Вставить("Результат", "ОК");
         СтруктураОтвета.Вставить("ТЗ", ОбъектXDTO);
         
      КонецЕсли;
      
   Иначе
      
      СтруктураОтвета = Новый Структура;
      СтруктураОтвета.Вставить("Результат", "Ошибка запроса к базе");
      СтруктураОтвета.Вставить("ТЗ", Новый ТаблицаЗначений);
      
   КонецЕсли;
   
   УстановитьПривилегированныйРежим(Ложь);
      
   Возврат СтруктураОтвета;
