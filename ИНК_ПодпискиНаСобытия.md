Процедура Документы_ОбработкаПроведения(Источник, Отказ, РежимПроведения) Экспорт
	
	Если ТипЗнч(Источник) = Тип("ДокументОбъект.ПриходныйОрдерНаТовары") Тогда
		
		ПриходныйОрдерНаТовары_ПроведениеДокумента(Источник, Отказ, РежимПроведения);
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.ПоступлениеБезналичныхДенежныхСредств") Тогда
		
		ПоступлениеБезналичныхДенежныхСредств_ПроведениеДокумента(Источник, Отказ, РежимПроведения);	
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.КоммерческоеПредложениеКлиенту") Тогда
		
		КоммерческоеПредложениеКлиенту_ПроведениеДокумента(Источник, Отказ, РежимПроведения);	
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.СписаниеБезналичныхДенежныхСредств") Тогда
		
		СписаниеБезналичныхДенежныхСредств_ПроведениеДокумента(Источник, Отказ, РежимПроведения);	
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.БИТ_ПредЗаказПоставщику") Тогда
		
		ДополнительныеРасходы_ДобавитьДокументКРасчету(Источник);	
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.БИТ_ПредПриобретениеТоваровУслуг") Тогда	
		
		ДополнительныеРасходы_ДобавитьДокументКРасчету(Источник);
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.РасчетСебестоимостиТоваров") Тогда	
		
		СебестоимостьТоваров_ДобавитьДокументКРасчету(Источник);		
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.ТаможеннаяДекларацияИмпорт") Тогда
		
		ТаможеннаяДекларацияИмпорт_ПроведениеДокумента(Источник, Отказ, РежимПроведения);   
		
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.РеализацияТоваровУслуг") Тогда
		
		РеализацияТоваровУслуг_ПроведениеДокумента(Источник, Отказ, РежимПроведения);   

	КонецЕсли;
	
КонецПроцедуры

Процедура Документы_ПриЗаписи(Источник, Отказ) Экспорт
	
	Если ТипЗнч(Источник) = Тип("ДокументОбъект.КоммерческоеПредложениеКлиенту") Тогда
		КоммерческоеПредложениеКлиенту_ЗаписьДокумента(Источник, Отказ);	
	ИначеЕсли ТипЗнч(Источник) = Тип("ДокументОбъект.ЗаказКлиента") Тогда
		ЗаказКлиента_ЗаписьДокумента(Источник, Отказ);
	КонецЕсли;
	
КонецПроцедуры

Процедура КоммерческоеПредложениеКлиенту_ЗаписьДокумента(Источник, Отказ) Экспорт
	
	Если НЕ Источник.ОбменДанными.Загрузка Тогда
		
		Если Источник.Дата >= Дата("20201111") Тогда
			
			Если Источник.Статус <> Перечисления.СтатусыКоммерческихПредложенийКлиентам.Отменено И 
				Источник.Статус <> Перечисления.СтатусыКоммерческихПредложенийКлиентам.НеСогласовано Тогда
				
				СтатусКП = РегистрыСведений.ИНК_ИсторияСтатусовКоммерческогоПредложения.ПолучитьПоследнийСтатусПоКП(Источник.Ссылка);			
				
				Если СтатусКП <> Источник.ИНК_СтатусКП И ЗначениеЗаполнено(Источник.ИНК_СтатусКП) Тогда
					
					МенеджерЗаписи = РегистрыСведений.ИНК_ИсторияСтатусовКоммерческогоПредложения.СоздатьМенеджерЗаписи();
					МенеджерЗаписи.Период 									= ТекущаяДата();
					МенеджерЗаписи.КоммерческоеПредложение   = Источник.Ссылка;
					МенеджерЗаписи.Статус 	=	Источник.ИНК_СтатусКП;
					
					Если Источник.ИНК_СтатусКП <> Перечисления.ИНК_СтатусыКП.ВыставленСчет И Источник.ИНК_СтатусКП <> Перечисления.ИНК_СтатусыКП.ОтказКлиента Тогда
						ГрафикРаботы = Справочники.Календари.НайтиПоНаименованию("Российская Федерация");		
						МенеджерЗаписи.КрайнийСрок = КалендарныеГрафики.ДатаПоКалендарю(ГрафикРаботы, НачалоДня(ТекущаяДата()), 3, Ложь);
					КонецЕсли;
					
					МенеджерЗаписи.Записать();
					
				КонецЕсли;
			КонецЕсли;
		КонецЕсли; 
		
		
	КонецЕсли;

Процедура ЗаказКлиента_ЗаписьДокумента(Источник, Отказ) 
	
	Если НЕ Источник.ОбменДанными.Загрузка Тогда

			Квоты = Источник.БИТ_Квотирование; 
			
			Для Каждого ТекущаяСтрока Из Квоты Цикл  
				
				Если ТекущаяСтрока.Комментарий <> "" Тогда
					РегистрыСведений.ИНК_КомментарииЗакупкиПоКвотам.ЗаписатьКомментарийВРегистр(
					ТекущаяСтрока.Номенклатура,

					Источник.Назначение,
					ТекущаяСтрока.Поставщик,
