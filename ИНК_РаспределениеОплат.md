// Общий модуль: ИНК_РаспределениеОплатПоЭтапамСервер
//
// Назначение: расчёт и распределение оплат по этапам графика и реализациям
//             в рамках Заказа клиента.
//
// Версия алгоритма: 3.0 (ТЗ v7)
// Платформа: 1С:Предприятие 8.3.23
// Конфигурация: УправлениеПредприятием 2.4.14.181
//
// Точки входа:
//   ВыполнитьОтложенныйРасчет(СсылкаНаОбъект)   — обычный режим (п.6.1 ТЗ)
//   ЗакрытьПериодПоОплатам(ЗаказКлиента, ПериодНачало, ПериодКонец) — закрытие периода (п.6.2 ТЗ)
//
// Зависимости:
//   ИНК_ДиагностикаРаспределенияОплат — модуль диагностики
//   ОбщегоНазначения (БСП) — ЗначениеРеквизитаОбъекта
//
// Изменения v7 (относительно v6):
//   - Фаза 2 упрощена: переизбыток → чистый аванс (без распределения по РТУ/этапам)
//   - Убрана ПоследняяРеализация — не нужна для Фазы 2
//   - Пропорциональное распределение: добавлены шаги 7.3-7.5 (ограничение + добивка копеек)
//   - Фиксация периода: критерий по дате Реализации (п.5.4 ТЗ)
//   - Защита п.5.5: проверка фиксаций по оплате перед расчётом

#Область Публичный

// Точка входа для обычного режима (п.6.1 ТЗ).
// Вызывается из подписки на событие после проведения/отмены проведения
// документов оплаты (ПБДС) и реализации (РТУ).
//
// Параметры:
//   СсылкаНаОбъект - ДокументСсылка - документ, по которому сработал триггер
//
Процедура ВыполнитьОтложенныйРасчет(СсылкаНаОбъект) Экспорт
	
	ЗаказыДляРасчета = ПолучитьМассивЗаказовИзОбъекта(СсылкаНаОбъект);
	
	Для Каждого ТекЗаказ Из ЗаказыДляРасчета Цикл
		Блокировка = Новый БлокировкаДанных;
		Элемент = Блокировка.Добавить("РегистрСведений.ИНК_РаспределениеОплатПоЭтапам");
		Элемент.УстановитьЗначение("ЗаказКлиента", ТекЗаказ);
		
		НачатьТранзакцию();
		Попытка
			Блокировка.Заблокировать();
			РаспределитьОплатыПоЗаказу(ТекЗаказ);
			ЗафиксироватьТранзакцию();
		Исключение
			ОтменитьТранзакцию();
			ТекстОшибки = СтрШаблон("Ошибка при расчете заказа %1: %2", ТекЗаказ, ОписаниеОшибки());
			ЗаписьЖурналаРегистрации("ИНК.РасчетАвансов", УровеньЖурналаРегистрации.Ошибка, , , ТекстОшибки);
		КонецПопытки;
	КонецЦикла;
	
КонецПроцедуры


// Точка входа для закрытия периода по оплатам (п.6.2 ТЗ).
// Вызывается бухгалтером из обработки с параметрами периода.
//
// Параметры:
//   ЗаказКлиента - ДокументСсылка.ЗаказКлиента или Неопределено (все заказы)
//   ПериодНачало - Дата - начало закрываемого периода
//   ПериодКонец  - Дата - конец закрываемого периода (последний день месяца)
//
Процедура ЗакрытьПериодПоОплатам(ЗаказКлиента = Неопределено, ПериодНачало, ПериодКонец) Экспорт
	
	ЗаказыДляЗакрытия = ПолучитьЗаказыДляЗакрытияПериода(ЗаказКлиента, ПериодКонец);
	
	Для Каждого ТекЗаказ Из ЗаказыДляЗакрытия Цикл
		
		Блокировка = Новый БлокировкаДанных;
		Элемент = Блокировка.Добавить("РегистрСведений.ИНК_РаспределениеОплатПоЭтапам");
		Элемент.УстановитьЗначение("ЗаказКлиента", ТекЗаказ);
		
		НачатьТранзакцию();
		Попытка
			Блокировка.Заблокировать();
			
			// v7: при закрытии периода выполняем перераспределение по всем отобранным заказам
			РаспределитьОплатыПоЗаказу(ТекЗаказ, Неопределено, ПериодКонец);
			
			// Фиксируем незакрытые записи распределений по реализациям закрываемого периода. Распределения будущих периодов не фиксируются. Переизбытки не фиксируются.
			ЗафиксироватьАвтозаписиДоДаты(ТекЗаказ, ПериодКонец);
			
			ЗафиксироватьТранзакцию();
		Исключение
			ОтменитьТранзакцию();
			ТекстОшибки = СтрШаблон(
				"Ошибка при закрытии периода для заказа %1: %2",
				ТекЗаказ, ОписаниеОшибки());
			ЗаписьЖурналаРегистрации(
				"ИНК.ЗакрытиеПериодаОплат",
				УровеньЖурналаРегистрации.Ошибка, , , ТекстОшибки);
		КонецПопытки;
		
	КонецЦикла;
	
КонецПроцедуры


#КонецОбласти




#Область СлужебныйПрограммныйИнтерфейс

// Основная процедура расчёта распределения оплат по одному заказу (п.6.3.3 ТЗ).
// В обычном режиме вызывается без параметров ПериодНачало/ПериодКонец.
// В режиме закрытия периода — с заполненными параметрами.
//
// Параметры:
//   ЗаказСсылка  - ДокументСсылка.ЗаказКлиента
//   ПериодНачало - Дата - заполнено только при закрытии периода (п.5.3 ТЗ)
//   ПериодКонец  - Дата - заполнено только при закрытии периода (п.5.3 ТЗ)
//
Процедура РаспределитьОплатыПоЗаказу(ЗаказСсылка, НачалоПериода, ПериодКонец)
	
	// Диагностика
	КонтекстДиаг = ИНК_ДиагностикаРаспределенияОплат.НачатьДиагностику(
		ЗаказСсылка,
		"Регламентный",
		"3.0", // v7: версия алгоритма
		ПараметрыСеанса.ТекущийПользователь
	);
	
	СуммаЗаказа = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ЗаказСсылка, "СуммаДокумента");
	Если СуммаЗаказа = 0 Тогда
		ИНК_ДиагностикаРаспределенияОплат.ЗавершитьДиагностику(КонтекстДиаг);
		Возврат;
	КонецЕсли;
	
	// 1) Данные
	ЭтапыПлан  = ПолучитьТаблицуЭтапов(ЗаказСсылка);
	ОплатыФакт = ПолучитьОплатыПоЗаказу(ЗаказСсылка);
	Реализации = ПолучитьРеализацииПоЗаказу(ЗаказСсылка);
	
	// Детерминированность — сортировка "Дата, Номер, Ссылка" (п.4.6 ТЗ)
	Если ОплатыФакт <> Неопределено И ОплатыФакт.Количество() > 0 Тогда
		ОплатыФакт.Сортировать("Дата, Номер, Ссылка");
	КонецЕсли;
	
	Если Реализации <> Неопределено И Реализации.Количество() > 0 Тогда
		Реализации.Сортировать("Дата, Номер, Ссылка");
	КонецЕсли;
	
	// v7 п.5.3: при закрытии периода — фильтруем оплаты до конца периода
	Если ПериодКонец <> Неопределено И ОплатыФакт <> Неопределено Тогда
		ОплатыДляРасчета = Новый Массив;
		Для Каждого СтрОп Из ОплатыФакт Цикл
			Если СтрОп.Дата <= КонецДня(ПериодКонец) Тогда
				ОплатыДляРасчета.Добавить(СтрОп);
			КонецЕсли;
		КонецЦикла;
	Иначе
		ОплатыДляРасчета = ОплатыФакт;
	КонецЕсли;
	
	// 2) Фиксации
	ДанныеФиксации = ПолучитьИтогиФиксацииПоЗаказу(ЗаказСсылка);
	
	// Проверки фиксаций (диагностика)
	ПроверитьПревышениеПлановыхЛимитовФиксациями(ЭтапыПлан, Реализации, ДанныеФиксации, КонтекстДиаг);
	
	ТаблицаИтог = СоздатьТаблицуДвижений();
	
	// Кэши авто-распределений ТЕКУЩЕГО пересчёта
	АвтоПоРТУ    = Новый Соответствие; // Реализация -> сумма авто (все этапы)
	АвтоПоЭтапам = Новый Соответствие; // УИДЭтапа -> сумма авто
	АвтоЭтапРТУ  = Новый Соответствие; // УИДЭтапа -> Соответствие(РТУ->Сумма авто)
	
	///////////////////////////////////////////////////////////////////////////
	// Шаг 0. БуферХвостов (п.4.4.4 ТЗ)
	// v7: убрана колонка УИДЦелевогоЭтапа — Фаза 2 не распределяет по этапам
	///////////////////////////////////////////////////////////////////////////
	БуферХвостов = Новый ТаблицаЗначений;
	БуферХвостов.Колонки.Добавить("ДокументОплаты");
	БуферХвостов.Колонки.Добавить("ДатаОплаты");
	БуферХвостов.Колонки.Добавить("СуммаХвоста", Новый ОписаниеТипов("Число", Новый КвалификаторыЧисла(15, 2)));
	
	///////////////////////////////////////////////////////////////////////////
	// Шаг 1а. Определяем ЦелевойЭтап для каждой оплаты (п.4.4.2 ТЗ)
	///////////////////////////////////////////////////////////////////////////
	
	// Строим накопленные пороги этапов
	НакопленныйПорог = 0;
	НакопленныеПорогиЭтапов = Новый Массив; // Массив Структур {УИДЭтапа, НакопленныйПорог}
	Для Каждого СтрЭтап Из ЭтапыПлан Цикл
		НакопленныйПорог = НакопленныйПорог + СтрЭтап.СуммаЭтапа;
		НакопленныеПорогиЭтапов.Добавить(Новый Структура(
			"УИДЭтапа, НакопленныйПорог",
			СтрЭтап.УИДЭтапа,
			НакопленныйПорог));
	КонецЦикла;
	СуммаВсехЭтапов = НакопленныйПорог;
	
	// Последний этап — для п.5 (ЦелевойЭтап = последний при переизбытке)
	СтрПоследнийЭтап = ?(ЭтапыПлан.Количество() > 0,
		ЭтапыПлан[ЭтапыПлан.Количество() - 1],
		Неопределено);
	
	// Вычисляем ЦелевойЭтап для каждой оплаты (Шаг 1а — СКОРРЕКТИРОВАННАЯ ФОРМУЛА v7.1)
	// 
	// Гибридная формула:
	// 1) ПервыйНезакрытый = первый этап с порогом > НакопленоДо
	// 2) ПорогПредыдущего = порог этапа перед ПервымНезакрытым (0 если первый)
	// 3) Если НакопленоДо > ПорогПредыдущего → ДОБИВКА: целевой по НакопленоПосле
	// 4) Если НакопленоДо <= ПорогПредыдущего → НОВАЯ ОПЛАТА: целевой = ПервыйНезакрытый
	//
	// Бизнес-смысл:
	// - Если оплата приходит на этап, который ещё не начат → целевой = этот этап,
	//   всё сверх лимита = переизбыток (чистый аванс)
	// - Если оплата добивает частично закрытый этап → штатное распределение,
	//   остаток переходит на следующие этапы (не переизбыток)
	//
	// Соответствие: ДокОплаты -> СтрокаЭтапа
	ЦелевыеЭтапыОплат = Новый Соответствие;
	НакопленоДо = 0;
	
	// Строим индекс ЭтапыПлан по УИДЭтапа для быстрого поиска
	ИндексЭтаповПоУИД = Новый Соответствие;
	Для Каждого СтрЭтап Из ЭтапыПлан Цикл
		ИндексЭтаповПоУИД.Вставить(СтрЭтап.УИДЭтапа, СтрЭтап);
	КонецЦикла;
	
	Для Каждого СтрОп Из ОплатыФакт Цикл
		
		НакопленоПосле = НакопленоДо + СтрОп.СуммаПоЗаказу;
		НайденЦелевойЭтап = Неопределено;
		
		// Шаг 1: находим ПервыйНезакрытый — первый этап с порогом > НакопленоДо
		ПервыйНезакрытыйИнд = -1;
		Для Инд = 0 По НакопленныеПорогиЭтапов.ВГраница() Цикл
			Если НакопленныеПорогиЭтапов[Инд].НакопленныйПорог > НакопленоДо Тогда
				ПервыйНезакрытыйИнд = Инд;
				Прервать;
			КонецЕсли;
		КонецЦикла;
		
		// Если все этапы закрыты — целевой = последний
		Если ПервыйНезакрытыйИнд = -1 Тогда
			НайденЦелевойЭтап = СтрПоследнийЭтап;
		Иначе
			// Шаг 2: ПорогПредыдущего
			ПорогПредыдущего = ?(ПервыйНезакрытыйИнд = 0,
				0,
				НакопленныеПорогиЭтапов[ПервыйНезакрытыйИнд - 1].НакопленныйПорог);
			
			// Шаг 3: определяем режим
			Если НакопленоДо > ПорогПредыдущего Тогда
				// ДОБИВКА: этап уже начат предыдущими оплатами
				// Целевой = первый этап с порогом >= НакопленоПосле
				Если НакопленоПосле > СуммаВсехЭтапов Тогда
					НайденЦелевойЭтап = СтрПоследнийЭтап;
				Иначе
					Для Каждого ИнфоЭтапа Из НакопленныеПорогиЭтапов Цикл
						Если ИнфоЭтапа.НакопленныйПорог >= НакопленоПосле Тогда
							НайденЦелевойЭтап = ИндексЭтаповПоУИД[ИнфоЭтапа.УИДЭтапа];
							Прервать;
						КонецЕсли;
					КонецЦикла;
				КонецЕсли;
			Иначе
				// НОВАЯ ОПЛАТА: этап ещё не начат
				// Целевой = ПервыйНезакрытый
				НайденЦелевойЭтап = ИндексЭтаповПоУИД[
					НакопленныеПорогиЭтапов[ПервыйНезакрытыйИнд].УИДЭтапа];
			КонецЕсли;
		КонецЕсли;
		
		ЦелевыеЭтапыОплат.Вставить(СтрОп.Ссылка, НайденЦелевойЭтап);
		НакопленоДо = НакопленоПосле;
		
	КонецЦикла;
	
	///////////////////////////////////////////////////////////////////////////
	// ФАЗА 1: распределение каждой оплаты по этапам до целевого включительно
	// (п.4.4 ТЗ v7)
	//
	// v7 отличия от v6:
	//   - Пропорциональное распределение: шаги 7.3-7.5 (ограничение + добивка)
	//   - Защита п.5.5: пропуск оплат с фиксациями (закрытый период)
	///////////////////////////////////////////////////////////////////////////
	
	Для Каждого СтрОп Из ОплатыДляРасчета Цикл
		
		// v7 п.5.5: защита от перерасчёта оплат закрытого периода
		// Если по оплате есть фиксации — значит период закрыт, пропускаем
		ЗафиксированоПоОплате = ДанныеФиксации.Оплаты[СтрОп.Ссылка];
		ЗафиксированоПоОплате = ?(ЗафиксированоПоОплате = Неопределено, 0, ЗафиксированоПоОплате);
		
		Если ЗафиксированоПоОплате > 0 Тогда
			// v7 п.5.5: фиксации есть — обычный пересчёт авто-записей по этой оплате запрещён
			ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(
				КонтекстДиаг,
				"D020_ПропускЗакрытойОплаты",
				"Информация",
				СтрШаблон("Оплата %1 имеет фиксации (%2 руб.) — пропущена при обычном пересчёте (п.5.5 ТЗ).",
					СтрОп.Ссылка, ЗафиксированоПоОплате),
				ЗафиксированоПоОплате,
				"", СтрОп.Ссылка, Неопределено);
			Продолжить;
		КонецЕсли;
		
		// Шаг 3 (п.4.4.2 ТЗ): ОстатокПлатежа = СуммаОплаты − фиксации по этой оплате
		ОстатокПлатежа = СтрОп.СуммаПоЗаказу;
		// v7: фиксации уже = 0 (проверено выше в обычном режиме),
		// но при закрытии периода оплата может иметь фиксации из прошлых периодов
		ОстатокПлатежа = ОстатокПлатежа - ЗафиксированоПоОплате;
		
		Если ОстатокПлатежа <= 0 Тогда
			Продолжить;
		КонецЕсли;
		
		ЦелевойЭтап = ЦелевыеЭтапыОплат[СтрОп.Ссылка];
		
		Если ЦелевойЭтап = Неопределено Тогда
			Продолжить;
		КонецЕсли;
		
		// Шаг 4 (п.4.4.2 ТЗ): итерация по этапам ДО целевого включительно
		Для Каждого СтрЭтап Из ЭтапыПлан Цикл
			
			Если ОстатокПлатежа <= 0 Тогда
				Прервать;
			КонецЕсли;
			
			// Шаг 4.1: ёмкость этапа = Лимит - Фиксации - АвтоТекПересчёта (п.4.2 ТЗ)
			ФиксацииВЭтапе = ?(ДанныеФиксации.Этапы[СтрЭтап.УИДЭтапа] = Неопределено,
				0, ДанныеФиксации.Этапы[СтрЭтап.УИДЭтапа]);
			АвтоВЭтапе = ?(АвтоПоЭтапам[СтрЭтап.УИДЭтапа] = Неопределено,
				0, АвтоПоЭтапам[СтрЭтап.УИДЭтапа]);
			ЕмкостьЭтапа = Макс(СтрЭтап.СуммаЭтапа - ФиксацииВЭтапе - АвтоВЭтапе, 0);
			
			// Шаг 4.2: сумма на этап
			СуммаНаЭтап = Мин(ОстатокПлатежа, ЕмкостьЭтапа);
			
			Если СуммаНаЭтап > 0 Тогда
				
				// Шаги 5-11: распределяем СуммаНаЭтап по РТУ внутри этапа
				РТУВЭтапе = СформироватьСписокДоступныхРТУВЭтапе(
					Реализации, СтрЭтап, ДанныеФиксации, АвтоПоРТУ, АвтоЭтапРТУ);
				
				Если РТУВЭтапе.База > 0 Тогда
					// v7 Шаги 7.1-7.5: пропорциональное распределение с ограничением и добивкой
					РаспределеноПоРТУ = ВыполнитьПропорциональноеРаспределение(
						ТаблицаИтог,
						ЗаказСсылка,
						СтрОп.Ссылка,
						СтрОп.Дата,
						СтрЭтап,
						РТУВЭтапе.Список,
						РТУВЭтапе.База,
						СуммаНаЭтап,
						АвтоПоРТУ,
						АвтоПоЭтапам,
						АвтоЭтапРТУ);
					
					// Шаги 9-11: остаток этапа, не вместившийся в РТУ → строка ожидания
					ОстатокЭтапаКРаспределению = Макс(0, Окр(СуммаНаЭтап - РаспределеноПоРТУ, 2));
					
					// v7.1: Коррекция копеечной погрешности округления.
					// Если остаток мал (≤ 0.01 × КолвоРТУ) — это артефакт того, что
					// round(СуммаЗаказа×%) ≠ Σ round(СуммаРТУ_i×%).
					// Вместо строки ожидания добавляем копейки к последней РТУ по сортировке.
					Если ОстатокЭтапаКРаспределению > 0
						И ОстатокЭтапаКРаспределению <= 0.01 * РТУВЭтапе.Список.Количество() Тогда
						
						// Ищем последнюю запись в ТаблицаИтог по этому этапу и оплате с заполненной Реализацией
						ИндексПоследней = -1;
						Для Сч = 0 По ТаблицаИтог.Количество() - 1 Цикл
							СтрИтог = ТаблицаИтог[Сч];
							Если СтрИтог.ДокументОплаты = СтрОп.Ссылка
								И СтрИтог.УИДЭтапа = СтрЭтап.УИДЭтапа
								И ЗначениеЗаполнено(СтрИтог.Реализация) Тогда
								ИндексПоследней = Сч;
							КонецЕсли;
						КонецЦикла;
						
						Если ИндексПоследней >= 0 Тогда
							// Добавляем копейки к последней РТУ
							СтрКоррекция = ТаблицаИтог[ИндексПоследней];
							СтрКоррекция.Сумма = СтрКоррекция.Сумма + ОстатокЭтапаКРаспределению;
							
							// Обновляем кэши
							ДобавитьСуммуВСоответствие(АвтоПоРТУ, СтрКоррекция.Реализация, ОстатокЭтапаКРаспределению);
							ДобавитьСуммуВСоответствие(АвтоПоЭтапам, СтрЭтап.УИДЭтапа, ОстатокЭтапаКРаспределению);
							ДобавитьСуммуЭтапРТУ(АвтоЭтапРТУ, СтрЭтап.УИДЭтапа, СтрКоррекция.Реализация, ОстатокЭтапаКРаспределению);
							
							ОстатокЭтапаКРаспределению = 0;
						КонецЕсли;
					КонецЕсли;
					
					Если ОстатокЭтапаКРаспределению > 0 Тогда
						// Шаг 11: строка ожидания реализаций (без Реализации, с УИДЭтапа)
						НоваяЗапись = ТаблицаИтог.Добавить();
						НоваяЗапись.ЗаказКлиента   = ЗаказСсылка;
						НоваяЗапись.Реализация     = Документы.РеализацияТоваровУслуг.ПустаяСсылка();
						НоваяЗапись.ДокументОплаты = СтрОп.Ссылка;
						НоваяЗапись.УИДЭтапа       = СтрЭтап.УИДЭтапа;
						НоваяЗапись.Сумма          = ОстатокЭтапаКРаспределению;
						НоваяЗапись.ТипОперации    = Перечисления.ИНК_ТипыОпераций.Аванс;
						НоваяЗапись.Зафиксировано  = Ложь;
						НоваяЗапись.ВыгруженоВБП   = Ложь;
						НоваяЗапись.ЭтоПереизбыток = Ложь;
						
						// Обновляем кэш этапа (строка ожидания занимает ёмкость, п.4.2 ТЗ)
						ДобавитьСуммуВСоответствие(АвтоПоЭтапам, СтрЭтап.УИДЭтапа, ОстатокЭтапаКРаспределению);
						
						ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(
							КонтекстДиаг,
							"D006_Ожидание_ОстатокВЭтапе",
							"Информация",
							СтрШаблон("Строка ожидания реализаций: остаток %1 в этапе %2 не вместился в доступные РТУ.",
								ОстатокЭтапаКРаспределению, СтрЭтап.УИДЭтапа),
							ОстатокЭтапаКРаспределению,
							СтрЭтап.УИДЭтапа, СтрОп.Ссылка, Неопределено);
					КонецЕсли;
					
				Иначе
					// Шаг 6 (п.4.4.2 ТЗ): нет доступных РТУ — строка ожидания реализаций
					НоваяЗапись = ТаблицаИтог.Добавить();
					НоваяЗапись.ЗаказКлиента   = ЗаказСсылка;
					НоваяЗапись.Реализация     = Документы.РеализацияТоваровУслуг.ПустаяСсылка();
					НоваяЗапись.ДокументОплаты = СтрОп.Ссылка;
					НоваяЗапись.УИДЭтапа       = СтрЭтап.УИДЭтапа;
					НоваяЗапись.Сумма          = СуммаНаЭтап;
					НоваяЗапись.ТипОперации    = Перечисления.ИНК_ТипыОпераций.Аванс;
					НоваяЗапись.Зафиксировано  = Ложь;
					НоваяЗапись.ВыгруженоВБП   = Ложь;
					НоваяЗапись.ЭтоПереизбыток = Ложь;
					
					// Обновляем кэш этапа
					ДобавитьСуммуВСоответствие(АвтоПоЭтапам, СтрЭтап.УИДЭтапа, СуммаНаЭтап);
					
					ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(
						КонтекстДиаг,
						"D006_Ожидание_НетРТУВЭтапе",
						"Информация",
						СтрШаблон("Строка ожидания реализаций: нет доступных РТУ в этапе %1. Сумма=%2.",
							СтрЭтап.УИДЭтапа, СуммаНаЭтап),
						СуммаНаЭтап,
						СтрЭтап.УИДЭтапа, СтрОп.Ссылка, Неопределено);
				КонецЕсли;
				
				// Шаг 11.1: уменьшаем остаток на ВСЮ сумму этапа
				ОстатокПлатежа = ОстатокПлатежа - СуммаНаЭтап;
				
			КонецЕсли; // СуммаНаЭтап > 0
			
			// Шаг 12 (п.4.4.2 ТЗ): завершаем итерацию на целевом этапе
			Если СтрЭтап.УИДЭтапа = ЦелевойЭтап.УИДЭтапа Тогда
				Прервать;
			КонецЕсли;
			
		КонецЦикла; // по этапам
		
		// Шаг 12 (п.4.4.4 ТЗ): остаток после целевого → БуферХвостов
		Если ОстатокПлатежа > 0 Тогда
			СтрБуфер = БуферХвостов.Добавить();
			СтрБуфер.ДокументОплаты = СтрОп.Ссылка;
			СтрБуфер.ДатаОплаты     = СтрОп.Дата;
			СтрБуфер.СуммаХвоста    = ОстатокПлатежа;
		КонецЕсли;
		
	КонецЦикла; // по оплатам — конец Фазы 1
	
	///////////////////////////////////////////////////////////////////////////
	// ФАЗА 2: обработка БуфераХвостов — запись чистого аванса (п.4.5 ТЗ v7)
	//
	// v7: ПОЛНОСТЬЮ ПЕРЕРАБОТАНА.
	// Переизбыток записывается как чистый аванс:
	//   Реализация = ПустаяСсылка(), УИДЭтапа = "", ЭтоПереизбыток = Истина.
	// Распределение по РТУ и этапам НЕ выполняется.
	// Пользователь вручную решает, куда отнести переизбыток.
	///////////////////////////////////////////////////////////////////////////
	
	Для Каждого СтрБуфер Из БуферХвостов Цикл
		
		Если СтрБуфер.СуммаХвоста <= 0 Тогда
			Продолжить;
		КонецЕсли;
		
		// п.4.5.2 ТЗ: запись чистого аванса
		НоваяЗапись = ТаблицаИтог.Добавить();
		НоваяЗапись.ЗаказКлиента   = ЗаказСсылка;
		НоваяЗапись.Реализация     = Документы.РеализацияТоваровУслуг.ПустаяСсылка();

		НоваяЗапись.УИДЭтапа       = ""; // пустая строка — отличает от строки ожидания
		НоваяЗапись.Сумма          = СтрБуфер.СуммаХвоста;
		НоваяЗапись.ТипОперации    = Перечисления.ИНК_ТипыОпераций.Аванс;
		НоваяЗапись.Зафиксировано  = Ложь;
		НоваяЗапись.ВыгруженоВБП   = Ложь;
		НоваяЗапись.ЭтоПереизбыток = Истина;
		
		ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(
			КонтекстДиаг,
			"D005_ЧистыйАванс",
			"Информация",
			СтрШаблон("Переизбыток оплаты %1 записан как чистый аванс: %2 руб.",
				СтрБуфер.ДокументОплаты, СтрБуфер.СуммаХвоста),
			СтрБуфер.СуммаХвоста,
			"", СтрБуфер.ДокументОплаты, Неопределено);
		
	КонецЦикла; // по БуферХвостов — конец Фазы 2
	
	///////////////////////////////////////////////////////////////////////////
	// Перенос флага ВыгруженоВБП: сохраняем Истина если сумма не изменилась

	ПеренестиФлагВыгруженоВБП(ЗаказСсылка, ТаблицаИтог);

// Возвращает итоги фиксаций по заказу из регистра (Зафиксировано = Истина).
// Без изменений относительно v6.
//
Функция ПолучитьИтогиФиксацииПоЗаказу(ЗаказСсылка)
	
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ
	|	РС.УИДЭтапа КАК УИДЭтапа,
	|	РС.Реализация КАК Реализация,
	|	РС.ДокументОплаты КАК ДокументОплаты,
	|	СУММА(РС.Сумма) КАК Сумма
	|ИЗ
	|	РегистрСведений.ИНК_РаспределениеОплатПоЭтапам КАК РС
	|ГДЕ
	|	РС.ЗаказКлиента = &Заказ
	|	И РС.Зафиксировано = ИСТИНА
	|СГРУППИРОВАТЬ ПО
	|	РС.УИДЭтапа,
	|	РС.Реализация,
	|	РС.ДокументОплаты";
	
	Запрос.УстановитьПараметр("Заказ", ЗаказСсылка);
	Результат = Запрос.Выполнить().Выгрузить();
	
	Данные = Новый Структура;
	Данные.Вставить("Этапы",      Новый Соответствие);
	Данные.Вставить("Реализации", Новый Соответствие);
	Данные.Вставить("Оплаты",     Новый Соответствие);
	Данные.Вставить("ОплатаЭтапРТУ", Новый Соответствие);
	Данные.Вставить("ЭтапРТУ",       Новый Соответствие);
	
	Для Каждого Стр Из Результат Цикл
		
		// 1. Итоги по этапам
		ТекСуммаЭтапа = Данные.Этапы[Стр.УИДЭтапа];
		ТекСуммаЭтапа = ?(ТекСуммаЭтапа = Неопределено, 0, ТекСуммаЭтапа);
		Данные.Этапы.Вставить(Стр.УИДЭтапа, ТекСуммаЭтапа + Стр.Сумма);
		
		// 2. Итоги по реализациям — общий контроль лимита РТУ
		Если ЗначениеЗаполнено(Стр.Реализация) Тогда
			ТекСуммаРТУ = Данные.Реализации[Стр.Реализация];
			ТекСуммаРТУ = ?(ТекСуммаРТУ = Неопределено, 0, ТекСуммаРТУ);
			Данные.Реализации.Вставить(Стр.Реализация, ТекСуммаРТУ + Стр.Сумма);
		КонецЕсли;
		
		// 3. Итоги по оплатам
		ТекСуммаОп = Данные.Оплаты[Стр.ДокументОплаты];
		ТекСуммаОп = ?(ТекСуммаОп = Неопределено, 0, ТекСуммаОп);
		Данные.Оплаты.Вставить(Стр.ДокументОплаты, ТекСуммаОп + Стр.Сумма);
		
		// 4. Фиксации по связке ДокОплаты -> Этап -> РТУ
		Если ЗначениеЗаполнено(Стр.Реализация) Тогда
			СоотвЭтапы = Данные.ОплатаЭтапРТУ[Стр.ДокументОплаты];
			Если СоотвЭтапы = Неопределено Тогда
				СоотвЭтапы = Новый Соответствие;
				Данные.ОплатаЭтапРТУ.Вставить(Стр.ДокументОплаты, СоотвЭтапы);
			КонецЕсли;
			
			СоотвРТУ = СоотвЭтапы[Стр.УИДЭтапа];
			Если СоотвРТУ = Неопределено Тогда
				СоотвРТУ = Новый Соответствие;
				СоотвЭтапы.Вставить(Стр.УИДЭтапа, СоотвРТУ);
			КонецЕсли;
			
			ТекФиксация = СоотвРТУ[Стр.Реализация];
			ТекФиксация = ?(ТекФиксация = Неопределено, 0, ТекФиксация);
			СоотвРТУ.Вставить(Стр.Реализация, ТекФиксация + Стр.Сумма);
			
			// 5. Фиксации по связке Этап -> РТУ
			ДобавитьСуммуЭтапРТУ(Данные.ЭтапРТУ, Стр.УИДЭтапа, Стр.Реализация, Стр.Сумма);
		КонецЕсли;
		
	КонецЦикла;
	
	Возврат Данные;

// Формирует список РТУ доступных для распределения в данном этапе.
// Без изменений относительно v6.
//
Функция СформироватьСписокДоступныхРТУВЭтапе(Реализации, СтрЭтап, ДанныеФиксации, АвтоПоРТУ, АвтоЭтапРТУ)
	
	Список = Новый Массив;
	База   = 0;
	
	Для Каждого СтрРеал Из Реализации Цикл
		
		// Общий остаток по РТУ
		ФиксацииПоРТУ = ДанныеФиксации.Реализации[СтрРеал.Ссылка];
		ФиксацииПоРТУ = ?(ФиксацииПоРТУ = Неопределено, 0, ФиксацииПоРТУ);
		
		АвтоПоЭтойРТУ = АвтоПоРТУ[СтрРеал.Ссылка];
		АвтоПоЭтойРТУ = ?(АвтоПоЭтойРТУ = Неопределено, 0, АвтоПоЭтойРТУ);
		
		ОстатокПоРТУОбщий = Макс(СтрРеал.СуммаДокумента - ФиксацииПоРТУ - АвтоПоЭтойРТУ, 0);
		Если ОстатокПоРТУОбщий <= 0 Тогда
			Продолжить;
		КонецЕсли;
		
		// Лимит РТУ в текущем этапе (п.4.2 ТЗ)
		ЛимитРТУВЭтапе = Окр(СтрРеал.СуммаДокумента * СтрЭтап.ПроцентЭтапа / 100, 2);
		
		// Уже распределено в этапе по этой РТУ (фикс + авто текущего пересчёта)
		ФиксацииЭтапРТУ   = ПолучитьСуммуЭтапРТУ(ДанныеФиксации.ЭтапРТУ, СтрЭтап.УИДЭтапа, СтрРеал.Ссылка);
		АвтоЭтапРТУ_Сумма = ПолучитьСуммуЭтапРТУ(АвтоЭтапРТУ, СтрЭтап.УИДЭтапа, СтрРеал.Ссылка);
		
		УжеРаспределено = ФиксацииЭтапРТУ + АвтоЭтапРТУ_Сумма;
		
		ОстатокРТУВЭтапе = Макс(ЛимитРТУВЭтапе - УжеРаспределено, 0);
		
		// Итоговое ограничение: минимум (этаповый лимит; общий остаток)
		Остаток = Мин(ОстатокРТУВЭтапе, ОстатокПоРТУОбщий);
		
		Если Остаток > 0 Тогда
			Список.Добавить(Новый Структура("Ссылка, Остаток, Дата", СтрРеал.Ссылка, Остаток, СтрРеал.Дата));
			База = База + Остаток;
		КонецЕсли;
		
	КонецЦикла;
	
	Возврат Новый Структура("Список, База", Список, База);

// v7: ПЕРЕРАБОТАНА. Пропорциональное распределение суммы по списку РТУ
// с ограничением и добивкой копеек (шаги 7.1-7.5 п.4.4.2 ТЗ).
//
// Шаг 7.1: рассчитать базу (СуммаОстатковРТУ)
// Шаг 7.2: для каждой РТУ вычислить долю и предварительную сумму
// Шаг 7.3: ограничить сверху остатком лимита
// Шаг 7.4: рассчитать недораспределённый остаток
// Шаг 7.5: добивка по 0.01 по убыванию доступного остатка
//
// Параметры:
//   ТаблицаИтог       - ТаблицаЗначений - таблица для накопления записей
//   ЗаказСсылка       - ДокументСсылка.ЗаказКлиента
//   ДокументОплаты    - ДокументСсылка - документ оплаты
//   ДатаОплаты        - Дата - дата оплаты (для определения ТипОперации)
//   СтрЭтап           - Структура - строка этапа
//   СписокСвободныхРТУ - Массив - список доступных РТУ {Ссылка, Остаток, Дата}
//   БазаДляПропорции  - Число - сумма остатков всех РТУ (для пропорции)
//   СуммаКРаспределению - Число - сумма на этап (СуммаНаЭтап)
//   АвтоПоРТУ, АвтоПоЭтапам, АвтоЭтапРТУ - кэши
//
// Возвращает:
//   Число - итого распределённая сумма по РТУ
//
Функция ВыполнитьПропорциональноеРаспределение(
	ТаблицаИтог,
	ЗаказСсылка,
	ДокументОплаты,
	ДатаОплаты,
	СтрЭтап,
	СписокСвободныхРТУ,
	БазаДляПропорции,
	СуммаКРаспределению,
	АвтоПоРТУ,
	АвтоПоЭтапам,
	АвтоЭтапРТУ)
	
	КолРТУ = СписокСвободныхРТУ.Количество();
	Если КолРТУ = 0 Тогда
		Возврат 0;
	КонецЕсли;
	
	// --- Шаги 7.1-7.3: предварительный расчёт с ограничением ---
	// Массив промежуточных результатов {Индекс, СуммаРТУ, ОстатокЛимита, ДоступныйОстаток}
	ПромежуточныеРезультаты = Новый Массив;
	
	Для Инд = 0 По КолРТУ - 1 Цикл
		
		РТУИнфо = СписокСвободныхРТУ[Инд];
		
		// Шаг 7.2: предварительная сумма
		ДоляРТУ = РТУИнфо.Остаток / БазаДляПропорции;
		СуммаРТУ = Окр(СуммаКРаспределению * ДоляРТУ, 2);
		
		// Шаг 7.3: ограничение сверху
		Если СуммаРТУ > РТУИнфо.Остаток Тогда
			СуммаРТУ = РТУИнфо.Остаток;
		КонецЕсли;
		
		ДоступныйОстаток = РТУИнфо.Остаток - СуммаРТУ;
		
		ПромежуточныеРезультаты.Добавить(Новый Структура(
			"Индекс, СуммаРТУ, ДоступныйОстаток, Ссылка, Дата",
			Инд, СуммаРТУ, ДоступныйОстаток, РТУИнфо.Ссылка, РТУИнфо.Дата));
		
	КонецЦикла;
	
	// --- Шаг 7.4: недораспределённый остаток ---
	СуммаРаспределено = 0;
	Для Каждого ПР Из ПромежуточныеРезультаты Цикл
		СуммаРаспределено = СуммаРаспределено + ПР.СуммаРТУ;
	КонецЦикла;
	НедораспределеноПоЭтапу = Окр(СуммаКРаспределению - СуммаРаспределено, 2);
	
	// --- Шаг 7.5: добивка по 0.01 по убыванию ДоступногоОстатка ---
	Если НедораспределеноПоЭтапу > 0 Тогда
		
		// Сортируем по убыванию ДоступныйОстаток, при равенстве — по (Дата, Ссылка) РТУ
		// Используем пузырьковую сортировку (массив обычно мал: 3-10 РТУ)
		Для Ин = 0 По ПромежуточныеРезультаты.ВГраница() - 1 Цикл
			Для К = Ин + 1 По ПромежуточныеРезультаты.ВГраница() Цикл
				ПР_И = ПромежуточныеРезультаты[Ин];
				ПР_К = ПромежуточныеРезультаты[К];
				Поменять = Ложь;
				
				Если ПР_И.ДоступныйОстаток < ПР_К.ДоступныйОстаток Тогда
					Поменять = Истина;
				ИначеЕсли ПР_И.ДоступныйОстаток = ПР_К.ДоступныйОстаток Тогда
					// При равенстве — сортировка РТУ по (Дата, Ссылка)
					Если ПР_И.Дата > ПР_К.Дата
						ИЛИ (ПР_И.Дата = ПР_К.Дата И ПР_И.Ссылка > ПР_К.Ссылка) Тогда
						Поменять = Истина;
					КонецЕсли;
				КонецЕсли;
				
				Если Поменять Тогда
					ПромежуточныеРезультаты[Ин] = ПР_К;
					ПромежуточныеРезультаты[К] = ПР_И;
				КонецЕсли;
			КонецЦикла;
		КонецЦикла;
		
		// Добавляем по 0.01 пока не исчерпан недораспределённый остаток
		Пока НедораспределеноПоЭтапу > 0 Цикл
			Добавлено = Ложь;
			Для Каждого ПР Из ПромежуточныеРезультаты Цикл
				Если НедораспределеноПоЭтапу <= 0 Тогда
					Прервать;
				КонецЕсли;
				Если ПР.ДоступныйОстаток > 0 Тогда
					ПР.СуммаРТУ = ПР.СуммаРТУ + 0.01;
					ПР.ДоступныйОстаток = ПР.ДоступныйОстаток - 0.01;
					НедораспределеноПоЭтапу = Окр(НедораспределеноПоЭтапу - 0.01, 2);
					Добавлено = Истина;
				КонецЕсли;
			КонецЦикла;
			// Защита от бесконечного цикла: если ни одной РТУ не добавили — выходим
			Если НЕ Добавлено Тогда
				Прервать;
			КонецЕсли;
		КонецЦикла;
		
	КонецЕсли;
	
	// --- Шаг 8: записываем результат в ТаблицаИтог ---
	Распределено = 0;
	
	Для Каждого ПР Из ПромежуточныеРезультаты Цикл
		
		Если ПР.СуммаРТУ <= 0 Тогда
			Продолжить;
		КонецЕсли;
		
		НоваяЗапись = ТаблицаИтог.Добавить();
		НоваяЗапись.ЗаказКлиента   = ЗаказСсылка;
		НоваяЗапись.Реализация     = ПР.Ссылка;
		НоваяЗапись.ДокументОплаты = ДокументОплаты;
		НоваяЗапись.УИДЭтапа       = СтрЭтап.УИДЭтапа;
		НоваяЗапись.Сумма          = ПР.СуммаРТУ;
		НоваяЗапись.Зафиксировано  = Ложь;
		НоваяЗапись.ВыгруженоВБП   = Ложь;
		НоваяЗапись.ЭтоПереизбыток = Ложь;
		
		// Тип операции: сравнение дат оплаты и реализации (п.4.3 ТЗ)
		НоваяЗапись.ТипОперации = ?(ДатаОплаты < ПР.Дата,
			Перечисления.ИНК_ТипыОпераций.Аванс,
			Перечисления.ИНК_ТипыОпераций.ЗачетОплаты);
		
		Распределено = Распределено + ПР.СуммаРТУ;
		
		// Обновляем кэши авто-распределений
		ДобавитьСуммуВСоответствие(АвтоПоРТУ, ПР.Ссылка, ПР.СуммаРТУ);
		ДобавитьСуммуВСоответствие(АвтоПоЭтапам, СтрЭтап.УИДЭтапа, ПР.СуммаРТУ);
		ДобавитьСуммуЭтапРТУ(АвтоЭтапРТУ, СтрЭтап.УИДЭтапа, ПР.Ссылка, ПР.СуммаРТУ);
		
	КонецЦикла;
	
	Возврат Распределено;

			Превышение = Окр(ФиксацииВЭтапе - СтрЭтап.СуммаЭтапа, 2);
			ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(

				Превышение, СтрЭтап.УИДЭтапа);
		КонецЕсли;
		
	КонецЦикла;
	
	// (B) Контроль: фиксации по паре (РТУ × этап) > плановый лимит
	Для Каждого ПараЭтап Из ДанныеФиксации.ЭтапРТУ Цикл
		
		УИДЭтапа = ПараЭтап.Ключ;

				0, УИДЭтапа);
			Продолжить;
		КонецЕсли;
		
		Для Каждого ПараРТУ Из СоотвРТУ Цикл
			
			Реализация = ПараРТУ.Ключ;
			СуммаФиксацийРТУВЭтапе = ПараРТУ.Значение;
			
			СуммаРТУ = СуммаРТУПоСсылке.Получить(Реализация);
			Если СуммаРТУ = Неопределено Тогда
				ИНК_ДиагностикаРаспределенияОплат.ЗафиксироватьСитуацию(
					КонтекстДиаг,
					"D012_Фиксации_РеализацияНеНайдена",
					"Предупреждение",
					"Найдены фиксации по реализации, но реализация не найдена в списке реализаций по заказу.",
					0, УИДЭтапа, Неопределено, Реализация);
				Продолжить;
			КонецЕсли;
			
			ПлановыйЛимитРТУВЭтапе = Окр(СуммаРТУ * ПроцентЭтапа / 100, 2);
			
			Если СуммаФиксацийРТУВЭтапе > ПлановыйЛимитРТУВЭтапе + 0.01 Тогда

			КонецЕсли;
			
		КонецЦикла;
