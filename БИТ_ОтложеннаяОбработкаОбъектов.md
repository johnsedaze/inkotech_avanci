//Добавление объекта для отложенной обработки. Процедура добавляет данные в РС "БИТ_ОтложеннаяОбработкаОбъектов"
//
//Атрибуты:
//	Ссылка - ссылка на объект
//	ВидДействия - ПеречислениеСсылка.БИТ_ВидыДействийОтложеннойОбработки - вид отложенного действия
//	ПериодЗаписи - для правильной последовательности обработки/проведения документов.
//
Процедура ДобавитьОбъектДляОтложеннойОбработки(Ссылка, ВидДействия, ПериодЗаписи = Неопределено) Экспорт
	
	 УстановитьПривилегированныйРежим(Истина);

	Если ПериодЗаписи = Неопределено тогда
		ДатаЗаписи = ТекущаяДатаСеанса();
	Иначе
		ДатаЗаписи = ПериодЗаписи;
	КонецЕсли;	
	
	Если ТранзакцияАктивна() Тогда
		
		// заблокируем объект для отложенной обработки
		Блокировка = Новый БлокировкаДанных();
		ЭлементБлокировки = Блокировка.Добавить("РегистрСведений.БИТ_ОтложеннаяОбработкаОбъектов");
		ЭлементБлокировки.УстановитьЗначение("Объект"		, Ссылка);
		ЭлементБлокировки.УстановитьЗначение("ВидДействия"	, ВидДействия);
		ЭлементБлокировки.УстановитьЗначение("Период"		, ДатаЗаписи);
		ЭлементБлокировки.Режим = РежимБлокировкиДанных.Исключительный;
		Блокировка.Заблокировать();
		
	КонецЕсли;
	
	//////////////////////////////////////////////////////////////////////
	//зарегистрируем сами данные
	
	МенеджерЗаписи = РегистрыСведений.БИТ_ОтложеннаяОбработкаОбъектов.СоздатьМенеджерЗаписи();
	МенеджерЗаписи.Период 		= ДатаЗаписи;
	МенеджерЗаписи.Объект 		= Ссылка;
	МенеджерЗаписи.ВидДействия 	= ВидДействия;
	МенеджерЗаписи.Статус 		= Перечисления.БИТ_СтатусыОбработкиОтложенныхДействий.Ожидание;
	
	МенеджерЗаписи.Записать(Истина);
	
	 УстановитьПривилегированныйРежим(Ложь);

	
КонецПроцедуры//ДобавитьОбъектДляОтложеннойОбработки()

//Функция получает список объектов для отложенной обработки
//
//Атрибуты:
//	ВидДействия - ПеречислениеСсылка.БИТ_ВидыДействийОтложеннойОбработки - вид отложенного действия
//
//Возвращаемое значение
//	ТаблицаЗначений - таблица, содержащая объекты для обработки
//		- Ссылка
//		- ДатаРегистрации
//		- ОписаниеОшибки
//
Функция ПолучитьОбъектыДляОтложеннойОбработки(ВидДействия, Количество = 1000) Экспорт
	
	КоличествоСтрокой = Формат(Количество,"ЧГ=");
		
	Запрос = Новый Запрос;
	Запрос.Текст = 
	"ВЫБРАТЬ ПЕРВЫЕ " + КоличествоСтрокой + "	
	|	БИТ_ОтложеннаяОбработкаОбъектов.Объект КАК Ссылка,
	|	БИТ_ОтложеннаяОбработкаОбъектов.Период КАК ДатаРегистрации,
	|	"""" КАК ОписаниеОшибки
	|ИЗ
	|	РегистрСведений.БИТ_ОтложеннаяОбработкаОбъектов.СрезПоследних(, ВидДействия = &ВидДействия) КАК БИТ_ОтложеннаяОбработкаОбъектов
	|ГДЕ 
	|	БИТ_ОтложеннаяОбработкаОбъектов.Статус = ЗНАЧЕНИЕ(Перечисление.БИТ_СтатусыОбработкиОтложенныхДействий.Ожидание)";
	
	Запрос.УстановитьПараметр("ВидДействия", ВидДействия);
	
	Результат = Запрос.Выполнить();
	
	Возврат Результат.Выгрузить();		
	
КонецФункции//ПолучитьОбъектыДляОбработки()

//Процедура записывает результат выполнения отложенной обработки
//Атрибуты:
//	Ссылка - ссылка на объект
//	ВидДействия -  ПеречислениеСсылка.БИТ_ВидыДействийОтложеннойОбработки - вид отложенного действия
//  ОписаниеОшибки - Строка
//
// Обрабатывается всегда только одна запись,  
// более ранние записи пропускаются
//
Процедура ЗаписатьРезультатОтложеннойОбработки(Ссылка, ВидДействия, ДатаРегистрации, ОписаниеОшибки = "") Экспорт
	УстановитьПривилегированныйРежим(Истина);
	Запрос = Новый Запрос;
	Запрос.Текст = 
	"ВЫБРАТЬ
	|	БИТ_ОтложеннаяОбработкаОбъектов.Период КАК Период,
	|	БИТ_ОтложеннаяОбработкаОбъектов.Объект,
	|	БИТ_ОтложеннаяОбработкаОбъектов.ВидДействия
	|ИЗ
	|	РегистрСведений.БИТ_ОтложеннаяОбработкаОбъектов КАК БИТ_ОтложеннаяОбработкаОбъектов
	|ГДЕ
	|	БИТ_ОтложеннаяОбработкаОбъектов.Период <= &Период
	|	И БИТ_ОтложеннаяОбработкаОбъектов.Объект = &Объект
	|	И БИТ_ОтложеннаяОбработкаОбъектов.ВидДействия = &ВидДействия
	|	И БИТ_ОтложеннаяОбработкаОбъектов.Статус = ЗНАЧЕНИЕ(Перечисление.БИТ_СтатусыОбработкиОтложенныхДействий.Ожидание)
	|
	|УПОРЯДОЧИТЬ ПО
	|	Период";
	
	Запрос.УстановитьПараметр("Период"		, ДатаРегистрации);
	Запрос.УстановитьПараметр("Объект"		, Ссылка);
	Запрос.УстановитьПараметр("ВидДействия"	, ВидДействия);
	
	РезультатЗапроса = Запрос.Выполнить();
		
	Если РезультатЗапроса.Пустой() Тогда
		Возврат
	КонецЕсли;
		
	Выборка = РезультатЗапроса.Выбрать();
	
	Попытка
		//НачатьТранзакцию();
		
		Пока Выборка.Следующий() Цикл
			
			СтруктураОтбораРегистра = СтруктураИзмеренийДляОтбора(Выборка.Период, Выборка.Объект, Выборка.ВидДействия);	
			
			Если Выборка.Период = ДатаРегистрации Тогда 
				
				Если ПустаяСтрока(ОписаниеОшибки)  Тогда
					ЗаписатьВыполненоВОтложеннуюОбработку(СтруктураОтбораРегистра);	
				Иначе
					ЗаписатьОшибкуВОтложеннуюОбработку(СтруктураОтбораРегистра, ОписаниеОшибки);
				КонецЕсли;	
			Иначе
				// пропускаем как неактуальные
				ЗаписатьПропущеноВОтложеннуюОбработку(СтруктураОтбораРегистра);
			КонецЕсли;	
												
		КонецЦикла;
		
		//ЗафиксироватьТранзакцию();
	Исключение
		//ОтменитьТранзакцию();
		
		ПредставлениеОшибки = ПодробноеПредставлениеОшибки(ИнформацияОбОшибке());
		ТекстОшибки = НСтр("ru = 'Ошибка при записи результата отложенной обработки. Ссылка: %1, Действие: %2. Подробности: %3'");
		ТекстОшибки = СтроковыеФункцииКлиентСервер.ПодставитьПараметрыВСтроку(ТекстОшибки, Ссылка, ВидДействия,ПредставлениеОшибки);
					
		ЗаписьЖурналаРегистрации(НСтр("ru = 'Отложенная обработка объектов.Запись результата отложенной обработки'"),
			УровеньЖурналаРегистрации.Ошибка,,Ссылка, 
			ТекстОшибки);

		//БИТ_ОбщегоНазначения.СообщитьОбОшибке(ТекстОшибки);	
		
		ВызватьИсключение ТекстОшибки;
	КонецПопытки;	
			
КонецПроцедуры//ЗаписатьРезультатОтложеннойОбработки()

Функция СтруктураИзмеренийДляОтбора(Период, Объект, ВидДействия) Экспорт
	
	СтруктураИзмерений = Новый Структура();
	СтруктураИзмерений.Вставить("Период", Период);
	СтруктураИзмерений.Вставить("Объект", Объект);
	СтруктураИзмерений.Вставить("ВидДействия", ВидДействия);	
	
	Возврат СтруктураИзмерений;
	
КонецФункции

// ОбновитьДанныеОтложеннойОбработки
//  процедура обновляет данные в РС "БИТ_ОтложеннаяОбработкаОбъектов", перезаписывая РС новыми данными
// Параметры
//	СтруктураОтбораРегистра	- Структура - Содержит значение измерений регистра для отбора
//	СтруктураДанныхДляОбновления- Структура данных для обновления: измерений, ресурсов, или реквизитов регистра.
Процедура ОбновитьДанныеОтложеннойОбработки(СтруктураОтбораРегистра, СтруктураДанныхДляОбновления) Экспорт
	
	НаборЗаписей = РегистрыСведений.БИТ_ОтложеннаяОбработкаОбъектов.СоздатьНаборЗаписей();
	НаборЗаписей.Отбор.Период.Установить(СтруктураОтбораРегистра.Период);
	НаборЗаписей.Отбор.Объект.Установить(СтруктураОтбораРегистра.Объект);
	НаборЗаписей.Отбор.ВидДействия.Установить(СтруктураОтбораРегистра.ВидДействия); 
	
	НаборЗаписей.Прочитать();
	
	Если НаборЗаписей.Количество() > 0 Тогда 
		Для каждого ЭлементНабора Из НаборЗаписей Цикл
			
			ЗаполнитьЗначенияСвойств(ЭлементНабора, СтруктураДанныхДляОбновления);
			
		КонецЦикла;
		
		НаборЗаписей.Записать();
	КонецЕсли;	
		
КонецПроцедуры // ОбновитьДанныеЖурналаОбмена

// ЗаписатьВыполненоВОтложеннуюОбработку
//  процедура добавляет данные в РС "БИТ_ОтложеннаяОбработкаОбъектов" со статусом "Выполнено"
// Параметры
//	СтруктураОтбораРегистра	- Структура - Содержит данные для отбора записи регистра
//
Процедура ЗаписатьВыполненоВОтложеннуюОбработку(СтруктураОтбораРегистра) Экспорт
	
	СтруктураДанныхДляОбновления = Новый Структура();
	СтруктураДанныхДляОбновления.Вставить("ДатаОбработки",					ТекущаяДата());
	СтруктураДанныхДляОбновления.Вставить("Статус", 						Перечисления.БИТ_СтатусыОбработкиОтложенныхДействий.Выполнено);
	
	ОбновитьДанныеОтложеннойОбработки(СтруктураОтбораРегистра, СтруктураДанныхДляОбновления);	
	
КонецПроцедуры

// ЗаписатьПропущеноВОтложеннуюОбработку
//  процедура добавляет данные в РС "БИТ_ОтложеннаяОбработкаОбъектов" со статусом "Пропущено"
// Параметры
//	СтруктураОтбораРегистра	- Структура - Содержит данные для отбора записи регистра
//	ОписаниеОшибки - Строка - описание ошибки
//
Процедура ЗаписатьПропущеноВОтложеннуюОбработку(СтруктураОтбораРегистра, ОписаниеОшибки = "") Экспорт
	
	СтруктураДанныхДляОбновления = Новый Структура();
	СтруктураДанныхДляОбновления.Вставить("ДатаОбработки",					ТекущаяДата());
	СтруктураДанныхДляОбновления.Вставить("Статус", 						Перечисления.БИТ_СтатусыОбработкиОтложенныхДействий.Пропущено);
	СтруктураДанныхДляОбновления.Вставить("ОписаниеОшибки", 				ОписаниеОшибки);

	ОбновитьДанныеОтложеннойОбработки(СтруктураОтбораРегистра, СтруктураДанныхДляОбновления);	
	
КонецПроцедуры

// ЗаписатьОшибкуВОтложеннуюОбработку
//  процедура добавляет данные в РС "БИТ_ОтложеннаяОбработкаОбъектов" со статусом "Ошибка"
// Параметры
//	СтруктураОтбораРегистра	- Структура - Содержит данные для отбора записи регистра
//	ОписаниеОшибки - Строка - описание ошибки
//
Процедура ЗаписатьОшибкуВОтложеннуюОбработку(СтруктураОтбораРегистра, ОписаниеОшибки) Экспорт
	
	СтруктураДанныхДляОбновления = Новый Структура();
	СтруктураДанныхДляОбновления.Вставить("ДатаОбработки",					ТекущаяДата());
	СтруктураДанныхДляОбновления.Вставить("Статус", 						Перечисления.БИТ_СтатусыОбработкиОтложенныхДействий.Ошибка);
	СтруктураДанныхДляОбновления.Вставить("ОписаниеОшибки", 				ОписаниеОшибки);

	ОбновитьДанныеОтложеннойОбработки(СтруктураОтбораРегистра, СтруктураДанныхДляОбновления);	
	
КонецПроцедуры
